119 //-------------------------------------------------------------------------------
120 // Set bounding sphere based on set of points
121 //-------------------------------------------------------------------------------
122 void
123 IvBoundingSphere::Set( const IvVector3* points, unsigned int numPoints )
124 {
125  ASSERT( points );
126  // compute minimal and maximal bounds
127  IvVector3 min(points[0]), max(points[0]);
128  unsigned int i;
129  for ( i = 1; i < numPoints; ++i )
130  {
131  if (points[i].x < min.x)
132  min.x = points[i].x;
133  else if (points[i].x > max.x )
134  max.x = points[i].x;
135  if (points[i].y < min.y)
136  min.y = points[i].y;
137  else if (points[i].y > max.y )
138  max.y = points[i].y;
139  if (points[i].z < min.z)
140  min.z = points[i].z;
141  else if (points[i].z > max.z )
142  max.z = points[i].z;
143  }
144  // compute center and radius
145  mCenter = 0.5f*(min + max);
146  float maxDistance = ::DistanceSquared( mCenter, points[0] );
147  for ( i = 1; i < numPoints; ++i )
148  {
149  float dist = ::DistanceSquared( mCenter, points[i] );
150  if (dist > maxDistance)
151  maxDistance = dist;
152  }
153  mRadius = ::IvSqrt( maxDistance );
154 }

191 IvBoundingSphere::Intersect( const IvBoundingSphere& other ) const
192 {

195  IvVector3 centerDiff = other.mCenter - mCenter;

204 // @ IvBoundingSphere::Intersect()
205 // ---------------------------------------------------------------------------
206 // Determine intersection between sphere and line
207 //-----------------------------------------------------------------------------
208 bool
209 IvBoundingSphere::Intersect( const IvLine3& line ) const
210 {
211  // compute intermediate values
212  IvVector3 w = mCenter - line.GetOrigin();
213  float wsq = w.Dot(w);
214  float proj = w.Dot(line.GetDirection());
215  float rsq = mRadius*mRadius;
216  float vsq = line.GetDirection().Dot(line.GetDirection());
217 
218  // test length of difference vs. radius
219  return ( vsq*wsq - proj*proj <= vsq*rsq );
220 }

297 //----------------------------------------------------------------------------
298 // @ ::Merge()
299 // ---------------------------------------------------------------------------
300 // Merge two spheres together to create a new one
301 //-----------------------------------------------------------------------------
302 void
303 Merge( IvBoundingSphere& result,
304  const IvBoundingSphere& s0, const IvBoundingSphere& s1 )
305 {
306  // get differences between them
307  IvVector3 diff = s1.mCenter - s0.mCenter;
308  float distsq = diff.Dot(diff);
309  float radiusdiff = s1.mRadius - s0.mRadius;
310 
311  // if one sphere inside other
312  if ( distsq <= radiusdiff*radiusdiff )
313  {
314  if ( s0.mRadius > s1.mRadius )
315  result = s0;
316  else
317  result = s1;
318  return;
319  }
320 
321  // build new sphere
322  float dist = ::IvSqrt( distsq );
323  float radius = 0.5f*( s0.mRadius + s1.mRadius + dist );
324  IvVector3 center = s0.mCenter;
325  if (!IvIsZero( dist ))
326  center += ((radius-s0.mRadius)/dist)*diff;
327 
328  result.SetRadius( radius );
329  result.SetCenter( center );
330 
331 } // End of ::Merge()

334 //----------------------------------------------------------------------------
335 // @ IvBoundingSphere::ComputeCollision()
336 // ---------------------------------------------------------------------------
337 // Compute parameters for collision between sphere and sphere
338 //-----------------------------------------------------------------------------
339 bool
340 IvBoundingSphere::ComputeCollision( const IvBoundingSphere& other, IvVector3& collisionNormal,
341  IvVector3& collisionPoint, float& penetration ) const
342 {
343  // do sphere check
344  float radiusSum = mRadius + other.mRadius;
345  collisionNormal = other.mCenter - mCenter;
346  float distancesq = collisionNormal.LengthSquared();
347  // if distance squared < sum of radii squared, collision!
348  if ( distancesq <= radiusSum*radiusSum )
349  {
350  // handle collision
351 
352  // penetration is distance - radii
353  float distance = ::IvSqrt(distancesq);
354  penetration = radiusSum - distance;
355  collisionNormal.Normalize();
356 
357  // collision point is average of penetration
358  collisionPoint = 0.5f*(mCenter + mRadius*collisionNormal)
359  + 0.5f*(other.mCenter - other.mRadius*collisionNormal);
360 
361  return true;
362  }
363 
364  return false;
365 
366 } // End of ::ComputeCollision()
