66 //
67 // Loads an IvIndexedGeometry object from a file of the following format
68 //
69 // VertexCount
70 //
71 // Vertex positions X Y Z for each vertex
72 // ...
73 //
74 // Vertex normals X Y Z for each vertex
75 // ...
76 //
77 // Vertex colors R G B for each vertex
78 // ...
79 //
80 // Triangle index count (count of indices, not triangles, so this should be 3x
81 // the triangle count
82 //
83 // Triangle indices
84 // ...
85 //
86 //-------------------------------------------------------------------------------
87 bool IvIndexedGeometry::LoadFromStream(IvReader& in, IvCapsule& capsule)
88 {
89  // no memory leaks
90  ASSERT(mVertices == 0);
91  ASSERT(mIndices == 0);
92 
93  IvCNPVertex* dataPtr = 0;
94  IvVector3* tempPosition = 0;
95  UInt32* indexPtr = 0;
96 
97  // get number of vertices
98  UInt32 numVerts;
99  in >> numVerts;
100  if ( !in.good() )
101  goto error_exit;
102 
103  // read verts
104  mVertices = IvRenderer::mRenderer->GetResourceManager()->CreateVertexBuffer(kCNPFormat, numVerts,
105  nullptr, kDefaultUsage);
106  dataPtr = (IvCNPVertex*) mVertices->BeginLoadData();
107  tempPosition = new IvVector3[numVerts]; // save for capsule creation
108  for ( UInt32 i = 0; i < numVerts; ++i )
109  {
110  float x, y, z;
111  in >> x >> y >> z;
112  tempPosition[i].Set(x,y,z);
113  dataPtr[i].position = tempPosition[i];
114  if (!in.good() )
115  {
116  goto error_exit;
117  }
118  }
119 
120  // read normals
121  for ( UInt32 i = 0; i < numVerts; ++i )
122  {
123  float x, y, z;
124  in >> x >> y >> z;
125  dataPtr[i].normal.Set(x,y,z);
126  if (!in.good())
127  {
128  goto error_exit;
129  }
130  }
131 
132  // read colors
133  for ( UInt32 i = 0; i < numVerts; ++i )
134  {
135  float r, g, b;
136  in >> r >> g >> b;
137  dataPtr[i].color.mRed = UChar8(r*255);
138  dataPtr[i].color.mGreen = UChar8(g*255);
139  dataPtr[i].color.mBlue = UChar8(b*255);
140  dataPtr[i].color.mAlpha = 255;
141  if (!in.good())
142  {
143  goto error_exit;
144  }
145  }
146 
147  if (!mVertices->EndLoadData())
148  {
149  goto error_exit;
150  }
151  dataPtr = 0;
152 
153  // get number of indices
154  UInt32 numIndices;
155  in >> numIndices;
156  if (!in.good())
157  {
158  goto error_exit;
159  }
160 
161  // read indices
162  mIndices = IvRenderer::mRenderer->GetResourceManager()->CreateIndexBuffer(numIndices,
163  nullptr, kDefaultUsage);
164  indexPtr = static_cast<UInt32*>(mIndices->BeginLoadData());
165  for ( UInt32 i = 0; i < numIndices; ++i )
166  {
167  in >> indexPtr[i];
168  if (!in.good() )
169  {
170  goto error_exit;
171  }
172  }
173  if (!mIndices->EndLoadData())
174  {
175  goto error_exit;
176  }
177  indexPtr = 0;
178 
179  // initialize the model-space capsule to the vertex data
180  capsule.Set(tempPosition, numVerts);
181 
182  delete [] tempPosition;
183 
184  return true;
185 
186 error_exit:
187  // error cleanup case
188  delete [] tempPosition;
189 
190  if (dataPtr)
191  {
192  (void)mVertices->EndLoadData();
193  }
194  if (indexPtr)
195  {
196  (void)mIndices->EndLoadData();
197  }
198  FreeResources();
199 
200  return false;
201 }
