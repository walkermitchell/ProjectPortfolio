141 // @ IvLinear::Evaluate()
142 //-------------------------------------------------------------------------------
143 // Evaluate spline
144 //-------------------------------------------------------------------------------
145 IvVector3
146 IvLinear::Evaluate( float t )
147 {
148  // make sure data is valid
149  ASSERT( mCount >= 2 );
150  if ( mCount < 2 )
151  return IvVector3::origin;
152 
153  // handle boundary conditions
154  if ( t <= mTimes[0] )
155  return mPositions[0];
156  else if ( t >= mTimes[mCount-1] )
157  return mPositions[mCount-1];
158 
159  // find segment and parameter
160  unsigned int i;
161  for ( i = 0; i < mCount-1; ++i )
162  {
163  if ( t < mTimes[i+1] )
164  {
165  break;
166  }
167  }
168  float t0 = mTimes[i];
169  float t1 = mTimes[i+1];
170  float u = (t - t0)/(t1 - t0);
171 
172  // evaluate
173  return (1-u)*mPositions[i] + u*mPositions[i+1];
174 
175 } // End of IvLinear::Initialize()
