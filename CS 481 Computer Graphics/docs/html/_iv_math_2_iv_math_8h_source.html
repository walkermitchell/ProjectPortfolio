52 //-------------------------------------------------------------------------------
53 // @ IvSqrt()
54 //-------------------------------------------------------------------------------
55 // Return square root of x, or close approximation
56 //
57 // This routine is from Lomont, "Floating Point Tricks," _Game Gems 6_
58 // See also www.lomont.org
59 //-------------------------------------------------------------------------------
60 inline float IvSqrt( float val )
61 {
62 #if defined(IV_APPROXIMATION)
63  assert(val >= 0);
64  IntOrFloat workval;
65  workval.f = val;
66  workval.i -= 0x3f800000; // subtract 127 from biased exponent
67  workval.i >>= 1; // requires signed shift to preserve sign
68  workval.i += 0x3f800000; // rebias the new exponent
69  workval.i &= 0x7FFFFFFF; // remove sign bit
70  return workval.f;
71 #else
72  return sqrtf( val );
73 #endif
74 }

77 //-------------------------------------------------------------------------------
78 // @ IvRecipSqrt()
79 //-------------------------------------------------------------------------------
80 // Return 1/sqrt(x), or close approximation
81 //
82 // This routine is a classic approximation to rsqrt, attributed to
83 // Greg Walsh.
84 // The optimized values used are from http://rrrola.wz.cz/inv_sqrt.html
85 //-------------------------------------------------------------------------------
86 inline float IvRecipSqrt( float val )
87 {
88 #if defined(IV_APPROXIMATION)
89  IntOrFloat workval;
90  workval.f = val;
91  // initial guess y0 with magic number
92  workval.i = 0x5F1FFFF9 - (workval.i>>1);
93  // Newton step, increases accuracy
94  const float estimate_sq = workval.f*workval.f;
95  workval.f *= 0.703952253f*(2.38924456f-val*estimate_sq);
96  return workval.f;
97 #else
98  return 1.0f/sqrtf( val );
99 #endif
100 }
