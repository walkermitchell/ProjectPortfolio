74  glBufferData( GL_ELEMENT_ARRAY_BUFFER, numIndices*sizeof(GL_UNSIGNED_INT), data,
75  usage == kDynamicUsage ? GL_DYNAMIC_DRAW : GL_STATIC_DRAW );
