122 //-------------------------------------------------------------------------------
123 // @ IvRenderer::SetWorldMatrix()
124 //-------------------------------------------------------------------------------
125 // Sets the world matrix for the renderer
126 //-------------------------------------------------------------------------------
127 void IvRenderer::SetWorldMatrix(const IvMatrix44& matrix)
128 {
129  mWorldMat = matrix;
130  mWVPMat = mProjectionMat*mViewMat*mWorldMat;
131  IvMatrix33 worldMat3x3;
132  IvVector3 col0(mWorldMat(0,0), mWorldMat(1,0), mWorldMat(2,0));
133  IvVector3 col1(mWorldMat(0,1), mWorldMat(1,1), mWorldMat(2,1));
134  IvVector3 col2(mWorldMat(0,2), mWorldMat(1,2), mWorldMat(2,2));
135  worldMat3x3.SetColumns(col0, col1, col2);
136  mNormalMat.Rotation(Transpose(Inverse(worldMat3x3)));
137 }
