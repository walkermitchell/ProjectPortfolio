217 static void
218 mouseCallback(GLFWwindow* window, int button, int action, int mods)
219 {
220  if ( button == GLFW_MOUSE_BUTTON_LEFT )
221  {
222  if ( action == GLFW_PRESS )
223  {
224  double x, y;
225  glfwGetCursorPos(window, &x, &y);
226 
227  int width;
228  int height;
229  glfwGetFramebufferSize(window, &width, &height);
230 
231  int windowWidth;
232  int windowHeight;
233  glfwGetWindowSize(window, &windowWidth, &windowHeight);
234 
235  double scaleX = width/(float)windowWidth;
236  double scaleY = height/(float)windowHeight;
237 
238  IvGame::mGame->mEventHandler->OnMouseDown( (int)(scaleX*x), (int)(scaleY*y) );
239  }
240  else
241  {
242  IvGame::mGame->mEventHandler->OnMouseUp();
243  }
244  }
