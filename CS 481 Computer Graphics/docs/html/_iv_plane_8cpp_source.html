158 // @ IvPlane::Set()
159 // ---------------------------------------------------------------------------
160 // Sets the parameters
161 //-----------------------------------------------------------------------------
162 void
163 IvPlane::Set( float a, float b, float c, float d )
164 {
165  // normalize for cheap distance checks
166  float lensq = a*a + b*b + c*c;
167  // length of normal had better not be zero
168  ASSERT( !IvIsZero( lensq ) );
169 
170  // recover gracefully
171  if ( IvIsZero( lensq ) )
172  {
173  mNormal = IvVector3::xAxis;
174  mOffset = 0.0f;
175  }
176  else
177  {
178  float recip = IvRecipSqrt(lensq);
179  mNormal.Set( a*recip, b*recip, c*recip );
180  mOffset = d*recip;
181  }
182 
183 } // End of IvPlane::Set()

186 //----------------------------------------------------------------------------
187 // @ IvPlane::Set()
188 // ---------------------------------------------------------------------------
189 // Sets the parameters
190 //-----------------------------------------------------------------------------
191 void
192 IvPlane::Set( const IvVector3& p0, const IvVector3& p1, const IvVector3& p2 )
193 {
194  // get plane vectors
195  IvVector3 u = p1 - p0;
196  IvVector3 v = p2 - p0;
197  IvVector3 w = u.Cross(v);
198 
199  // normalize for cheap distance checks
200  float lensq = w.x*w.x + w.y*w.y + w.z*w.z;
201  // length of normal had better not be zero
202  ASSERT( !IvIsZero( lensq ) );
203 
204  // recover gracefully
205  if ( IvIsZero( lensq ) )
206  {
207  mNormal = IvVector3::xAxis;
208  mOffset = 0.0f;
209  }
210  else
211  {
212  float recip = 1.0f/lensq;
213  mNormal.Set( w.x*recip, w.y*recip, w.z*recip );
214  mOffset = -mNormal.Dot(p0);
215  }
216 
217 } // End of IvPlane::Set()

225 IvPlane
226 IvPlane::Transform( float scale, const IvQuat& rotate, const IvVector3& translate ) const
227 {
228  IvPlane plane;
229 
230  // get rotation matrix
231  IvMatrix33 rotmatrix(rotate);
232 
233  // transform to get normal
234  plane.mNormal = rotmatrix*mNormal/scale;
235 
236  // transform to get offset
237  IvVector3 newTrans = translate*rotmatrix;
238  plane.mOffset = -newTrans.Dot( mNormal )/scale + mOffset;
