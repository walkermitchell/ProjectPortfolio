162 //-------------------------------------------------------------------------------
163 // @ IvVertexBufferD3D11::Create()
164 //-------------------------------------------------------------------------------
165 // Create the resources for the vertex buffer
166 //-------------------------------------------------------------------------------
167 bool
168 IvVertexBufferD3D11::Create(IvVertexFormat format, unsigned int numVertices, void* data, IvDataUsage usage,
169  ID3D11Device* device)
170 {
171  if (numVertices == 0)
172  {
173  return false;
174  }
175 
176  if (usage == kImmutableUsage && !data)
177  {
178  return false;
179  }
180 
181  D3D11_BUFFER_DESC vertexBufferDesc;
182  vertexBufferDesc.ByteWidth = numVertices*kIvVFSize[format];
183  switch (usage)
184  {
185  default:
186  case kDefaultUsage:
187  vertexBufferDesc.Usage = D3D11_USAGE_DEFAULT;
188  break;
189  case kDynamicUsage:
190  vertexBufferDesc.Usage = D3D11_USAGE_DYNAMIC;
191  break;
192  case kImmutableUsage:
193  vertexBufferDesc.Usage = D3D11_USAGE_IMMUTABLE;
194  break;
195  }
196  mUsage = usage;
197  vertexBufferDesc.BindFlags = D3D11_BIND_VERTEX_BUFFER;
198  vertexBufferDesc.CPUAccessFlags = (usage == kDynamicUsage) ? D3D11_CPU_ACCESS_WRITE : 0;
199  vertexBufferDesc.MiscFlags = 0;
200  //*** replacement for D3DMANAGED?
201 
202  D3D11_SUBRESOURCE_DATA* initDataPtr = nullptr;
203  D3D11_SUBRESOURCE_DATA initData;
204  if (data)
205  {
206  initDataPtr = &initData;
207  initData.pSysMem = data;
208  initData.SysMemPitch = 0;
209  initData.SysMemSlicePitch = 0;
210  }
211 
212  if (FAILED(device->CreateBuffer(&vertexBufferDesc, initDataPtr, &mBufferPtr)))
213  {
214  mBufferPtr = 0;
215  return false;
216  }
217 
218  if (sInputLayout[format] == 0)
219  {
220  D3D11_INPUT_ELEMENT_DESC* elements = nullptr;
221  UINT numElements = 0;
222  const char* shaderString = nullptr;
223 
224  switch (format)
225  {
226  case kPFormat:
227  elements = sPFormatElements;
228  numElements = sizeof(sPFormatElements) / sizeof(D3D11_INPUT_ELEMENT_DESC);
229  shaderString = sDummyShaderPFormat;
230  break;
231  case kCPFormat:
232  elements = sCPFormatElements;
233  numElements = sizeof(sCPFormatElements) / sizeof(D3D11_INPUT_ELEMENT_DESC);
234  shaderString = sDummyShaderCPFormat;
235  break;
236  case kNPFormat:
237  elements = sNPFormatElements;
238  numElements = sizeof(sNPFormatElements) / sizeof(D3D11_INPUT_ELEMENT_DESC);
239  shaderString = sDummyShaderNPFormat;
240  break;
241  case kCNPFormat:
242  elements = sCNPFormatElements;
243  numElements = sizeof(sCNPFormatElements) / sizeof(D3D11_INPUT_ELEMENT_DESC);
244  shaderString = sDummyShaderCNPFormat;
245  break;
246  case kTCPFormat:
247  elements = sTCPFormatElements;
248  numElements = sizeof(sTCPFormatElements) / sizeof(D3D11_INPUT_ELEMENT_DESC);
249  shaderString = sDummyShaderTCPFormat;
250  break;
251  case kTNPFormat:
252  elements = sTNPFormatElements;
253  numElements = sizeof(sTNPFormatElements) / sizeof(D3D11_INPUT_ELEMENT_DESC);
254  shaderString = sDummyShaderTNPFormat;
255  break;
256  }
257 
258  if (nullptr == shaderString)
259  {
260  mBufferPtr->Release();
261  mBufferPtr = 0;
262  return false;
263  }
264 
265  ID3DBlob* code;
266  ID3DBlob* errorMessages = nullptr;
267 
268  // compile the shader to assembly
269  if (FAILED(D3DCompile(shaderString, strlen(shaderString) + 1, nullptr, nullptr, nullptr, "vs_main", "vs_4_0",
270  D3DCOMPILE_ENABLE_STRICTNESS|D3DCOMPILE_DEBUG, 0, &code, &errorMessages)))
271  {
272  const char* errors = reinterpret_cast<const char*>(errorMessages->GetBufferPointer());
273  DEBUG_OUT("Vertex shader error: ");
274  DEBUG_OUT(errors << std::endl);
275  errorMessages->Release();
276  mBufferPtr->Release();
277  mBufferPtr = 0;
278  return false;
279  }
280  if (errorMessages)
281  {
282  errorMessages->Release();
283  }
284 
285  if (FAILED(device->CreateInputLayout(elements, numElements, code->GetBufferPointer(), code->GetBufferSize(), &sInputLayout[format])))
286  {
287  mBufferPtr->Release();
288  mBufferPtr = 0;
289  code->Release();
290  return false;
291  }
292  code->Release();
293  }
294  else
295  {
296  sInputLayout[format]->AddRef();
297  }
298 
299  mVertexFormat = format;
300  mNumVertices = numVertices;
301 
302  return true;
303 }

360 void *
361 IvVertexBufferD3D11::BeginLoadData()
362 {
363  if (mUsage == kImmutableUsage)
364  {
365  return nullptr;
366  }
367 
368  // already "locked"
369  if (mDataPtr)
370  {
371  return nullptr;
372  }
373 
374  mDataPtr = (void*) new unsigned char[mNumVertices*kIvVFSize[mVertexFormat]];
375  return mDataPtr;

378 //-------------------------------------------------------------------------------
379 // @ IvVertexBufferD3D11::EndLoadData()
380 //-------------------------------------------------------------------------------
381 // Unlock the buffer, we're done loading
382 // Returns true if all went well
383 //-------------------------------------------------------------------------------
384 bool
385 IvVertexBufferD3D11::EndLoadData()
386 {
387  if (mUsage == kImmutableUsage)
388  {
389  return false;
390  }
391 
392  // not already "locked"
393  if (!mDataPtr)
394  {
395  return false;
396  }
397 
398  // this is seriously ugly -- not clear how to easily get the context down here
399  ID3D11DeviceContext* d3dContext = ((IvRendererD3D11*)IvRenderer::mRenderer)->GetContext();
400  if (kDefaultUsage == mUsage)
401  {
402  // use UpdateSubresource()
403  d3dContext->UpdateSubresource(mBufferPtr, 0, nullptr, mDataPtr, mNumVertices*kIvVFSize[mVertexFormat], mNumVertices*kIvVFSize[mVertexFormat]);
404  }
405  else if (kDynamicUsage == mUsage)
406  {
407  // use Map/Unmap
408  D3D11_MAPPED_SUBRESOURCE mappedResource;
409  ZeroMemory(&mappedResource, sizeof(D3D11_MAPPED_SUBRESOURCE));
410  // Disable GPU access to the vertex buffer data.
411  if (S_OK != d3dContext->Map(mBufferPtr, 0, D3D11_MAP_WRITE_DISCARD, 0, &mappedResource))
412  {
413  return false;
414  }
415 
416  void* bufferData = mappedResource.pData;
417  memcpy(bufferData, mDataPtr, mNumVertices*kIvVFSize[mVertexFormat]);
418 
419  // Reenable GPU access to the vertex buffer data.
420  d3dContext->Unmap(mBufferPtr, 0);
421  }
422 
423  delete[] mDataPtr;
424  mDataPtr = nullptr;
425 
426  return true;
427 }
