52 // @ IvVertexBufferOGL::Create()
53 //-------------------------------------------------------------------------------
54 // Create the resources for the vertex buffer
55 //-------------------------------------------------------------------------------
56 bool
57 IvVertexBufferOGL::Create(IvVertexFormat format, unsigned int numVertices,
58  void* data, IvDataUsage usage)
59 {
60  if ( numVertices == 0 || mBufferID != 0 )
61  return false;
62 
63  if ( usage == kImmutableUsage && !data )
64  {
65  return false;
66  }
67 
68  // create vertex array handle
69  glGenVertexArrays(1, &mVertexArrayID);
70  glBindVertexArray(mVertexArrayID);
71 
72  // create the handle
73  glGenBuffers( 1, &mBufferID );
74  glBindBuffer( GL_ARRAY_BUFFER, mBufferID );
75 
76  // allocate the memory
77  (void) glGetError(); // clear any previous errors (probably not safe)
78  glBufferData( GL_ARRAY_BUFFER, numVertices*kIvVFSize[format], data,
79  usage == kDynamicUsage ? GL_DYNAMIC_DRAW : GL_STATIC_DRAW );
80  if ( glGetError() != GL_NO_ERROR )
81  {
82  Destroy();
83  return false;
84  }
85 
86  mVertexFormat = format;
87  mNumVertices = numVertices;
88 
89  // set up vertex attributes
90  // this should match sShaderHeader in IvVertexShaderOGL.cpp
91 #define COLOR 0
92 #define NORMAL 1
93 #define TEXCOORD0 2
94 #define POSITION 3
95  size_t stride = kIvVFSize[mVertexFormat];
96  size_t offset = 0;
97  switch (mVertexFormat)
98  {
99  case kPFormat:
100  break;
101 
102  case kCPFormat:
103  default:
104  glEnableVertexAttribArray(COLOR);
105  glVertexAttribPointer(COLOR, 4, GL_UNSIGNED_BYTE, GL_TRUE, stride, (GLvoid*) offset);
106  offset += 4*sizeof(UChar8);
107  break;
108 
109  case kNPFormat:
110  glEnableVertexAttribArray(NORMAL);
111  glVertexAttribPointer(NORMAL, 3, GL_FLOAT, GL_FALSE, stride, (GLvoid*) offset);
112  offset += 3*sizeof(float);
113  break;
114 
115  case kCNPFormat:
116  glEnableVertexAttribArray(COLOR);
117  glVertexAttribPointer(COLOR, 4, GL_UNSIGNED_BYTE, GL_TRUE, stride, (GLvoid*) offset);
118  offset += 4*sizeof(UChar8);
119  glEnableVertexAttribArray(NORMAL);
120  glVertexAttribPointer(NORMAL, 3, GL_FLOAT, GL_FALSE, stride, (GLvoid*) offset);
121  offset += 3*sizeof(float);
122  break;
123 
124  case kTCPFormat:
125  glEnableVertexAttribArray(TEXCOORD0);
126  glVertexAttribPointer(TEXCOORD0, 2, GL_FLOAT, GL_FALSE, stride, (GLvoid*) offset);
127  offset += 2*sizeof(float);
128  glEnableVertexAttribArray(COLOR);
129  glVertexAttribPointer(COLOR, 4, GL_UNSIGNED_BYTE, GL_TRUE, stride, (GLvoid*) offset);
130  offset += 4*sizeof(UChar8);
131  break;
132 
133  case kTNPFormat:
134  glEnableVertexAttribArray(TEXCOORD0);
135  glVertexAttribPointer(TEXCOORD0, 2, GL_FLOAT, GL_FALSE, stride, (GLvoid*) offset);
136  offset += 2*sizeof(float);
137  glEnableVertexAttribArray(NORMAL);
138  glVertexAttribPointer(NORMAL, 3, GL_FLOAT, GL_FALSE, stride, (GLvoid*) offset);
139  offset += 3*sizeof(float);
140  break;
141  }
142 
143  // we always do position
144  glEnableVertexAttribArray(POSITION);
145  glVertexAttribPointer(POSITION, 3, GL_FLOAT, GL_FALSE, stride, (GLvoid*) offset);
146 
147  glBindVertexArray( 0 );
148 
149  mUsage = usage;
150 
151  return true;
152 }
