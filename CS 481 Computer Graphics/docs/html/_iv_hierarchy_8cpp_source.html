99 // Add node data to the tree
100 //-------------------------------------------------------------------------------
101 bool IvHierarchy::AddNode(int index, unsigned char parent,
102  IvReader& inStream,
103  const IvVector3& xlate, const IvQuat& rotate, float scale)
104 {
105  if (mNumNodes == 0 || index >= mNumNodes)
106  {
107  return false;
108  }
109  // child nodes must be later in the array than their parents
110  if (!((index == 0 && parent == 0) || (parent < index)))
111  {
112  return false;
113  }
114  mParents[index] = parent;
115  mGeometries[index].LoadFromStream(inStream, mLocalCapsules[index]);
116  mLocalTransforms[index].mRotate = rotate;
117  mLocalTransforms[index].mTranslate = xlate;
118  mLocalTransforms[index].mScale = scale;
119 
120  return true;
121 }

123 //-------------------------------------------------------------------------------
124 // @ IvHierarchy::UpdateWorldTransforms()
125 //-------------------------------------------------------------------------------
126 // Propagate transforms down and back up the hierarchy
127 //-------------------------------------------------------------------------------
128 void IvHierarchy::UpdateWorldTransforms()
129 {
130  ASSERT(mNumNodes > 0);
131  ASSERT(mParents[0] == 0); // root node should be 0
132  mWorldTransforms[0] = mLocalTransforms[0];
133  mWorldTransforms[0].mRotate.Normalize();
134  mWorldCapsules[0] = mLocalCapsules[0].Transform(mWorldTransforms[0].mScale,
135  mWorldTransforms[0].mRotate,
136  mWorldTransforms[0].mTranslate);
137  mWorldSpheres[0].SetCenter(mWorldCapsules[0].GetSegment().GetCenter());
138  mWorldSpheres[0].SetRadius(mWorldCapsules[0].GetSegment().Length() * 0.5f
139  + mWorldCapsules[0].GetRadius());
140 
141  // propagate down to children
142  for (int i = 1; i < mNumNodes; ++i)
143  {
144  int parent = mParents[i];
145  ASSERT(parent < i);
146  // combine parent transform with our own
147  float parentScale = mWorldTransforms[parent].mScale;
148  mWorldTransforms[i].mTranslate =
149  mWorldTransforms[parent].mRotate.Rotate(mLocalTransforms[i].mTranslate);
150  mWorldTransforms[i].mTranslate *= parentScale;
151  mWorldTransforms[i].mTranslate += mWorldTransforms[parent].mTranslate;
152  mWorldTransforms[i].mScale = mLocalTransforms[i].mScale*parentScale;
153  mWorldTransforms[i].mRotate = mWorldTransforms[parent].mRotate*mLocalTransforms[i].mRotate;
154  mWorldTransforms[i].mRotate.Normalize();
155 
156  // transform local capsule into world space
157  mWorldCapsules[i] = mLocalCapsules[i].Transform(mWorldTransforms[i].mScale,
158  mWorldTransforms[i].mRotate,
159  mWorldTransforms[i].mTranslate);
160 
161  // generate world space bounding sphere
162  mWorldSpheres[i].SetCenter(mWorldCapsules[i].GetSegment().GetCenter());
163  mWorldSpheres[i].SetRadius(mWorldCapsules[i].GetSegment().Length() * 0.5f
164  + mWorldCapsules[i].GetRadius());
165  }
166 
167  // merge bounding spheres up to parents
168  for (int i = mNumNodes - 1; i > 0; --i)
169  {
170  int parent = mParents[i];
171  Merge(mWorldSpheres[parent], mWorldSpheres[parent], mWorldSpheres[i]);
172  }
173 
174 } // End of IvHierarchy::UpdateWorldTransforms

177 //-------------------------------------------------------------------------------
178 // @ IvHierarchy::Render()
179 //-------------------------------------------------------------------------------
180 // Draws the entire hierarchy
181 //-------------------------------------------------------------------------------
182 void IvHierarchy::Render()
183 {
184  for (int i = 0; i < mNumNodes; ++i)
185  {
186  // build 4x4 matrix
187  IvMatrix44 transform(mWorldTransforms[i].mRotate);
188  transform(0, 0) *= mWorldTransforms[i].mScale;
189  transform(1, 0) *= mWorldTransforms[i].mScale;
190  transform(2, 0) *= mWorldTransforms[i].mScale;
191  transform(0, 1) *= mWorldTransforms[i].mScale;
192  transform(1, 1) *= mWorldTransforms[i].mScale;
193  transform(2, 1) *= mWorldTransforms[i].mScale;
194  transform(0, 2) *= mWorldTransforms[i].mScale;
195  transform(1, 2) *= mWorldTransforms[i].mScale;
196  transform(2, 2) *= mWorldTransforms[i].mScale;
197  transform(0, 3) = mWorldTransforms[i].mTranslate.x;
198  transform(1, 3) = mWorldTransforms[i].mTranslate.y;
199  transform(2, 3) = mWorldTransforms[i].mTranslate.z;
200 
201  // set current local-to-world matrix
202  IvSetWorldMatrix(transform);
203 
204  mGeometries[i].Render();
205 
206  if (gDisplayLeafBounds)
207  {
208  IvRenderer::mRenderer->SetFillMode(kWireframeFill);
209 
210  IvSetWorldIdentity();
211  IvDrawCapsule(mWorldCapsules[i].GetSegment(), mWorldCapsules[i].GetRadius(), kOrange);
212  IvRenderer::mRenderer->SetFillMode(kSolidFill);
213  }
214 
215  if (gDisplayHierarchyBounds)
216  {
217  IvRenderer::mRenderer->SetFillMode(kWireframeFill);
218  IvMatrix44 ident;
219  ident.Identity();
220  ident(0, 3) = mWorldSpheres[i].GetCenter().x;
221  ident(1, 3) = mWorldSpheres[i].GetCenter().y;
222  ident(2, 3) = mWorldSpheres[i].GetCenter().z;
223 
224  IvSetWorldMatrix(ident);
225  IvDrawSphere(mWorldSpheres[i].GetRadius(), kYellow);
226 
227  IvRenderer::mRenderer->SetFillMode(kSolidFill);
228  }
229  }
230 
231 } // End of IvHierarchy::Render
