218 // @ IvDrawFloor()
219 //-------------------------------------------------------------------------------
220 // Draw tiled floor at the origin
221 //-------------------------------------------------------------------------------
222 void
223 IvDrawFloor()
224 {
225  // build data if not there
226  if ( floorVerts == 0 )
227  {
228  size_t currentOffset = IvStackAllocator::mScratchAllocator->GetCurrentOffset();
229  IvCPVertex* vertexPtr = (IvCPVertex*)IvStackAllocator::mScratchAllocator->Allocate(kIvVFSize[kCPFormat] * 4 * 81);
230  UInt32* indexPtr = (UInt32*)IvStackAllocator::mScratchAllocator->Allocate(sizeof(UInt32) * 6 * 81);
231  int currentVertex = 0;
232  int currentIndex = 0;
233  for ( int i = -12; i <= 12; i += 3 )
234  {
235  for ( int j = -12; j <= 12; j += 3 )
236  {
237  IvColor color;
238  if ( ( i & 0x1 ) == (j & 0x1 ))
239  color.Set( 64, 64, 64, 255 );
240  else
241  color.Set( 192, 192, 192, 255 );
242  vertexPtr[currentVertex].position.Set(0.0f + (float)i, 0.0f + (float)j, 0.0f);
243  vertexPtr[currentVertex].color = color;
244  vertexPtr[currentVertex+1].position.Set(3.0f + (float)i, 0.0f + (float)j, 0.0f);
245  vertexPtr[currentVertex+1].color = color;
246  vertexPtr[currentVertex+2].position.Set(0.0f + (float)i, 3.0f + (float)j, 0.0f);
247  vertexPtr[currentVertex+2].color = color;
248  vertexPtr[currentVertex+3].position.Set(3.0f + (float)i, 3.0f + (float)j, 0.0f);
249  vertexPtr[currentVertex+3].color = color;
250  indexPtr[currentIndex++] = currentVertex;
251  indexPtr[currentIndex++] = currentVertex+1;
252  indexPtr[currentIndex++] = currentVertex+2;
253  indexPtr[currentIndex++] = currentVertex+2;
254  indexPtr[currentIndex++] = currentVertex+1;
255  indexPtr[currentIndex++] = currentVertex+3;
256  currentVertex += 4;
257  }
258  }
259 
260  floorVerts = IvRenderer::mRenderer->GetResourceManager()->CreateVertexBuffer(kCPFormat, 4 * 81, vertexPtr,
261  kImmutableUsage);
262  floorIndices = IvRenderer::mRenderer->GetResourceManager()->CreateIndexBuffer(6 * 81, indexPtr,
263  kImmutableUsage);
264 
265  if (!floorVerts || !floorIndices)
266  {
267  if (floorVerts)
268  {
269  IvRenderer::mRenderer->GetResourceManager()->Destroy(floorVerts);
270  floorVerts = 0;
271  }
272  if (floorIndices)
273  {
274  IvRenderer::mRenderer->GetResourceManager()->Destroy(floorIndices);
275  floorIndices = 0;
276  }
277  }
278 
279  IvStackAllocator::mScratchAllocator->Reset(currentOffset);
280  }
281 
282  IvSetWorldIdentity();
283 
284  // clear to default shader
285  IvShaderProgram* oldShader = IvRenderer::mRenderer->GetShaderProgram();
286  IvRenderer::mRenderer->SetShaderProgram(0);
287 
288  // draw it
289  IvRenderer::mRenderer->Draw(kTriangleListPrim, floorVerts, floorIndices);
290 
291  // restore original shader
292  IvRenderer::mRenderer->SetShaderProgram(oldShader);
293 
294 } // End of IvDrawFloor()
