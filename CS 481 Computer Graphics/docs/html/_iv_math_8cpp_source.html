31 // One quadrant of a polynomial approximation of sin
32 //-------------------------------------------------------------------------------
33 static inline float IvPolynomialSinQuadrant(float a)
34 {
35 
36  // sin(a) ~ a - 0.16666 a^3 + 0.0083143 a^5 - 0.00018542 a^7
37 
38  return a * ( 1.0f + a * a * (-0.16666f
39  + a * a * (0.0083143f
40  - a * a * 0.00018542f)));
41 }

44 // @ IvFastSinCos()
45 //-------------------------------------------------------------------------------
46 // Returns the floating-point sine and cosine of the argument. This method uses
47 // a polynomial approximation
48 //-------------------------------------------------------------------------------
49 void IvFastSinCos( float a, float& sina, float& cosa )
50 {
51  // The concepts in this function are based on Robin Green's GDC 2003 session
52  // "Faster Math Functions". The slides are available online from Sony R&D at
53  // http://www.research.scea.com/gdc2003/fast-math-functions.html
54 
55  // work with a non-negative angle - sin is odd and cos is even, so this is
56  // easy to do
57  bool bNegate = false;
58  if (a < 0.0f)
59  {
60  a = -a;
61  bNegate = true;
62  }
63 
64  // The polynomial approximation requires that the angle "a" be decomposed as
65  // a = Pi/2 * intA + floatA,
66  // where intA is an integer and floatA is a float in the range [0, Pi/2]
67  // Because sin and cos have a period of 2Pi, we will use both intA and floatA
68  // to compute sina and cosa.
69  const float kTwoOverPI = 1.0f / kHalfPI;
70  float floatA = kTwoOverPI * a;
71  int intA = (int)floatA;
72 
73  // This next step (getting the floating-point remainder of floatA would
74  // normally involve subtracting two nearly equal values, which could lead
75  // to catastrophic cancelation.
76  // Instead, we break Pi/2 into two parts that add to Pi/2.
77  // One is a rational that single-precision can represent exactly,
78  // while the other is the remainder. We do the subtraction in two stages
79  // to avoid cancelling.
80  const float kRationalHalfPI = 201 / 128.0f;
81  const float kRemainderHalfPI = 4.8382679e-4f;
82 
83  floatA = (a - kRationalHalfPI * intA) - kRemainderHalfPI * intA;
84 
85  // same trick to slide the angle over by Pi/2
86  float floatAMinusHalfPi = (floatA - kRationalHalfPI) - kRemainderHalfPI;
87 
88  // need to switch, based on the quadrant of Pi, so we compute a fast "mod 4"
89  // of intA
90  switch (intA & 3)
91  {
92  case 0: // 0 - Pi/2
93  sina = IvPolynomialSinQuadrant(floatA);
94  cosa = IvPolynomialSinQuadrant(-floatAMinusHalfPi);
95  break;
96  case 1: // Pi/2 - Pi
97  sina = IvPolynomialSinQuadrant(-floatAMinusHalfPi);
98  cosa = IvPolynomialSinQuadrant(-floatA);
99  break;
100  case 2: // Pi - 3Pi/2
101  sina = IvPolynomialSinQuadrant(-floatA);
102  cosa = IvPolynomialSinQuadrant(floatAMinusHalfPi);
103  break;
104  case 3: // 3Pi/2 - 2Pi
105  sina = IvPolynomialSinQuadrant(floatAMinusHalfPi);
106  cosa = IvPolynomialSinQuadrant(floatA);
107  break;
108  };
109 
110  if (bNegate)
111  sina = -sina;
112 
113 } // End of IvFastSinCos
