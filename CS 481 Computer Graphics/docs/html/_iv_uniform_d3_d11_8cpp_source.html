127 // Float vec3 set
128 //-------------------------------------------------------------------------------
129 void IvUniformD3D11::SetValue(const IvVector3& value, unsigned int index)
130 {
131  if (mType != kFloat3Uniform)
132  return;
133 
134  mValue.mVector3[index] = value;
135 
136  // float3 array is packed as float4 array
137  char* offset = reinterpret_cast<char*>(mOffset);
138  IvVector3* currVec3 = mValue.mVector3;
139  for (unsigned int i = 0; i < mCount; ++i)
140  {
141  memcpy(offset, currVec3, sizeof(IvVector3));
142  ++currVec3;
143  offset += 4*sizeof(float);
144  }
145  mConstantTable->MarkDirty();
146 }
