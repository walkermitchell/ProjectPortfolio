277 void IvUniformOGL::Update()
278 {
279  if (mShader != IvRenderer::mRenderer->GetShaderProgram())
280  return;
281 
282  if ( mType == kTextureUniform )
283  {
284  mValue.mTexture->MakeActive(mTextureStage);
285  }
286 
287  if (!mNeedsUpdate)
288  return;
289 
290  switch (mType)
291  {
292  case kFloatUniform:
293  glUniform1fv(mShaderIndex, mCount, mValue.mFloat);
294  break;
295  case kFloat3Uniform:
296  glUniform3fv(mShaderIndex, mCount, (GLfloat*)mValue.mVector3);
297  break;
298  case kFloat4Uniform:
299  glUniform4fv(mShaderIndex, mCount, (GLfloat*)mValue.mVector4);
300  break;
301  case kFloatMatrix44Uniform:
302  glUniformMatrix4fv(mShaderIndex, mCount, GL_FALSE, (GLfloat*)mValue.mMatrix44);
303  break;
304  case kTextureUniform:
305  glUniform1i(mShaderIndex, mTextureStage);
306  break;
307  };
308 
309  mNeedsUpdate = false;
310
