34 // Get the constant table for this shader
35 //-------------------------------------------------------------------------------
36 IvConstantTableD3D11*
37 IvConstantTableD3D11::Create(ID3D11Device* device, ID3DBlob* code)
38 {
39  ID3D11ShaderReflection* pReflector = nullptr;
40  D3DReflect(code->GetBufferPointer(), code->GetBufferSize(),
41  IID_ID3D11ShaderReflection, (void**)&pReflector);
42 
43  // first get constant table
44  IvConstantTableD3D11* table = new IvConstantTableD3D11();
45 
46  ID3D11ShaderReflectionConstantBuffer* globalConstants =
47  pReflector->GetConstantBufferByName("$Globals");
48  D3D11_SHADER_BUFFER_DESC constantBufferDesc = { 0 };
49  HRESULT hr = globalConstants->GetDesc(&constantBufferDesc);
50  if (FAILED(hr))
51  {
52  // no constants in this shader
53  goto get_textures;
54  }
55 
56  table->mBacking = new char[constantBufferDesc.Size];
57  table->mBackingSize = constantBufferDesc.Size;
58  if (nullptr == table->mBacking)
59  {
60  delete table;
61  pReflector->Release();
62  return nullptr;
63  }
64 
65  for (unsigned int i = 0; i < constantBufferDesc.Variables; ++i)
66  {
67  ID3D11ShaderReflectionVariable* var = globalConstants->GetVariableByIndex(i);
68  D3D11_SHADER_VARIABLE_DESC varDesc;
69  if (FAILED(var->GetDesc(&varDesc)))
70  {
71  delete table;
72  pReflector->Release();
73  return nullptr;
74  }
75  ID3D11ShaderReflectionType* type = var->GetType();
76  D3D11_SHADER_TYPE_DESC typeDesc;
77  if (FAILED(type->GetDesc(&typeDesc)))
78  {
79  delete table;
80  pReflector->Release();
81  return nullptr;
82  }
83 
84  IvConstantDesc constantDesc;
85  constantDesc.mOffset = table->mBacking + varDesc.StartOffset;
86  if (typeDesc.Class == D3D_SVC_SCALAR && typeDesc.Type == D3D_SVT_FLOAT)
87  {
88  constantDesc.mType = IvUniformType::kFloatUniform;
89  }
90  else if (typeDesc.Class == D3D_SVC_VECTOR && typeDesc.Type == D3D_SVT_FLOAT
91  && typeDesc.Columns == 3)
92  {
93  constantDesc.mType = IvUniformType::kFloat3Uniform;
94  }
95  else if (typeDesc.Class == D3D_SVC_VECTOR && typeDesc.Type == D3D_SVT_FLOAT
96  && typeDesc.Columns == 4)
97  {
98  constantDesc.mType = IvUniformType::kFloat4Uniform;
99  }
100  else if ((typeDesc.Class == D3D_SVC_MATRIX_ROWS || typeDesc.Class == D3D_SVC_MATRIX_COLUMNS)
101  && typeDesc.Type == D3D_SVT_FLOAT
102  && typeDesc.Rows == 4 && typeDesc.Columns == 4)
103  {
104  constantDesc.mType = IvUniformType::kFloatMatrix44Uniform;
105  }
106  else
107  {
108  // unsupported uniform type
109  delete table;
110  pReflector->Release();
111  return nullptr;
112  }
113  constantDesc.mCount = typeDesc.Elements != 0 ? typeDesc.Elements : 1;
114 
115  table->mConstants[varDesc.Name] = constantDesc;
116  }
117 
118  D3D11_BUFFER_DESC cbDesc;
119  cbDesc.ByteWidth = constantBufferDesc.Size;
120  cbDesc.Usage = D3D11_USAGE_DYNAMIC;
121  cbDesc.BindFlags = D3D11_BIND_CONSTANT_BUFFER;
122  cbDesc.CPUAccessFlags = D3D11_CPU_ACCESS_WRITE;
123  cbDesc.MiscFlags = 0;
124  cbDesc.StructureByteStride = 0;
125 
126  if (FAILED(device->CreateBuffer(&cbDesc, nullptr, &table->mBuffer)))
127  {
128  delete table;
129  pReflector->Release();
130  return nullptr;
131  }
132 
133 get_textures:
134  // now get any textures and samplers
135  D3D11_SHADER_DESC shaderDesc = { 0 };
136  hr = pReflector->GetDesc(&shaderDesc);
137  if (FAILED(hr))
138  {
139  pReflector->Release();
140  return table;
141  }
142  int resourceCount = shaderDesc.BoundResources;
143 
144  for (int i = 0; i < resourceCount; ++i)
145  {
146  D3D11_SHADER_INPUT_BIND_DESC bindingDesc = { 0 };
147  pReflector->GetResourceBindingDesc(i, &bindingDesc);
148  if (bindingDesc.Type == D3D_SIT_TEXTURE && bindingDesc.ReturnType == D3D_RETURN_TYPE_FLOAT
149  && bindingDesc.Dimension == D3D_SRV_DIMENSION_TEXTURE2D && bindingDesc.NumSamples == -1
150  && bindingDesc.BindCount == 1)
151  {
152  IvConstantDesc& constantDesc = table->mConstants[bindingDesc.Name];
153  constantDesc.mType = IvUniformType::kTextureUniform;
154  constantDesc.mTextureSlot = bindingDesc.BindPoint;
155  }
156  else if (bindingDesc.Type == D3D_SIT_SAMPLER && bindingDesc.BindCount == 1)
157  {
158  char strippedName[256];
159  const char* samplerKeyword = strstr(bindingDesc.Name, "Sampler");
160  strncpy(strippedName, bindingDesc.Name, samplerKeyword - bindingDesc.Name);
161  strippedName[samplerKeyword - bindingDesc.Name] = '\0';
162  IvConstantDesc& constantDesc = table->mConstants[strippedName];
163  constantDesc.mType = IvUniformType::kTextureUniform;
164  constantDesc.mSamplerSlot = bindingDesc.BindPoint;
165  }
166  else if (bindingDesc.Type == D3D_SIT_CBUFFER && !_strnicmp(bindingDesc.Name, "$Globals", 8))
167  {
168  // ignore this, we handled it above
169  }
170  else
171  {
172  // unsupported resource type
173  delete table;
174  pReflector->Release();
175  return nullptr;
176  }
177  }
178 
179  pReflector->Release();
180 
181  return table;
182 }

209  if (constantDesc->mType == kTextureUniform)

222 // Make this constant buffer active for vertex shader
223 //-------------------------------------------------------------------------------
224 bool IvConstantTableD3D11::MakeActiveVS(ID3D11DeviceContext* context)
225 {
226  if (nullptr == mBuffer || nullptr == mBacking)
227  return false;
228 
229  if (mDirty)
230  {
231  // Lock the constant buffer so it can be written to.
232  D3D11_MAPPED_SUBRESOURCE mappedResource;
233  HRESULT result = context->Map(mBuffer, 0, D3D11_MAP_WRITE_DISCARD, 0, &mappedResource);
234  if (FAILED(result))
235  {
236  return false;
237  }
238 
239  // Copy the data
240  memcpy(mappedResource.pData, mBacking, mBackingSize);
241 
242  // Unlock the constant buffer.
243  context->Unmap(mBuffer, 0);
244 
245  mDirty = false;
246  }
247 
248  context->VSSetConstantBuffers(0, 1, &mBuffer);
249  return true;
250 }
