233 //----------------------------------------------------------------------------
234 // @ DistanceSquared()
235 // ---------------------------------------------------------------------------
236 // Returns the distance squared between two line segments.
237 // Based on article and code by Dan Sunday at www.geometryalgorithms.com
238 //-----------------------------------------------------------------------------
239 float
240 DistanceSquared( const IvLineSegment3& segment0, const IvLineSegment3& segment1,
241  float& s_c, float& t_c )
242 {
243  // compute intermediate parameters
244  IvVector3 w0 = segment0.mOrigin - segment1.mOrigin;
245  float a = segment0.mDirection.Dot( segment0.mDirection );
246  float b = segment0.mDirection.Dot( segment1.mDirection );
247  float c = segment1.mDirection.Dot( segment1.mDirection );
248  float d = segment0.mDirection.Dot( w0 );
249  float e = segment1.mDirection.Dot( w0 );
250 
251  float denom = a*c - b*b;
252  // parameters to compute s_c, t_c
253  float sn, sd, tn, td;
254 
255  // if denom is zero, try finding closest point on segment1 to origin0
256  if ( IvIsZero(denom) )
257  {
258  // clamp s_c to 0
259  sd = td = c;
260  sn = 0.0f;
261  tn = e;
262  }
263  else
264  {
265  // clamp s_c within [0,1]
266  sd = td = denom;
267  sn = b*e - c*d;
268  tn = a*e - b*d;
269 
270  // clamp s_c to 0
271  if (sn < 0.0f)
272  {
273  sn = 0.0f;
274  tn = e;
275  td = c;
276  }
277  // clamp s_c to 1
278  else if (sn > sd)
279  {
280  sn = sd;
281  tn = e + b;
282  td = c;
283  }
284  }
285 
286  // clamp t_c within [0,1]
287  // clamp t_c to 0
288  if (tn < 0.0f)
289  {
290  t_c = 0.0f;
291  // clamp s_c to 0
292  if ( -d < 0.0f )
293  {
294  s_c = 0.0f;
295  }
296  // clamp s_c to 1
297  else if ( -d > a )
298  {
299  s_c = 1.0f;
300  }
301  else
302  {
303  s_c = -d/a;
304  }
305  }
306  // clamp t_c to 1
307  else if (tn > td)
308  {
309  t_c = 1.0f;
310  // clamp s_c to 0
311  if ( (-d+b) < 0.0f )
312  {
313  s_c = 0.0f;
314  }
315  // clamp s_c to 1
316  else if ( (-d+b) > a )
317  {
318  s_c = 1.0f;
319  }
320  else
321  {
322  s_c = (-d+b)/a;
323  }
324  }
325  else
326  {
327  t_c = tn/td;
328  s_c = sn/sd;
329  }
330 
331  // compute difference vector and distance squared
332  IvVector3 wc = w0 + s_c*segment0.mDirection - t_c*segment1.mDirection;
333  return wc.Dot(wc);
334 
335 } // End of DistanceSquared()

487 //----------------------------------------------------------------------------
488 // @ IvLineSegment3::DistanceSquared()
489 // ---------------------------------------------------------------------------
490 // Returns the distance squared between line segment and point.
491 //-----------------------------------------------------------------------------
492 float
493 DistanceSquared( const IvLineSegment3& segment, const IvVector3& point,
494  float& t_c )
495 {
496  IvVector3 w = point - segment.mOrigin;
497  float proj = w.Dot(segment.mDirection);
498  // endpoint 0 is closest point
499  if ( proj <= 0 )
500  {
501  t_c = 0.0f;
502  return w.Dot(w);
503  }
504  else
505  {
506  float vsq = segment.mDirection.Dot(segment.mDirection);
507  // endpoint 1 is closest point
508  if ( proj >= vsq )
509  {
510  t_c = 1.0f;
511  return w.Dot(w) - 2.0f*proj + vsq;
512  }
513  // otherwise somewhere else in segment
514  else
515  {
516  t_c = proj/vsq;
517  return w.Dot(w) - t_c*proj;
518  }
519  }
520 
521 } // End of ::DistanceSquared()

779 // ---------------------------------------------------------------------------
780 // Returns the closest point on line segment to point
781 //-----------------------------------------------------------------------------
782 IvVector3 IvLineSegment3::ClosestPoint( const IvVector3& point ) const
783 {
784  IvVector3 w = point - mOrigin;
785  float proj = w.Dot(mDirection);
786  // endpoint 0 is closest point
787  if ( proj <= 0.0f )
788  return mOrigin;
789  else
790  {
791  float vsq = mDirection.Dot(mDirection);
792  // endpoint 1 is closest point
793  if ( proj >= vsq )

795  // else somewhere else in segment
