68 //-------------------------------------------------------------------------------
69 // @ IvQuat::IvQuat()
70 //-------------------------------------------------------------------------------
71 // Rotation matrix constructor
72 //-------------------------------------------------------------------------------
73 IvQuat::IvQuat( const IvMatrix33& rotation )
74 {
75  float trace = rotation.Trace();
76  if ( trace > 0.0f )
77  {
78  float s = ::IvSqrt( trace + 1.0f );
79  w = s*0.5f;
80  float recip = 0.5f/s;
81  x = (rotation(2,1) - rotation(1,2))*recip;
82  y = (rotation(0,2) - rotation(2,0))*recip;
83  z = (rotation(1,0) - rotation(0,1))*recip;
84  }
85  else
86  {
87  unsigned int i = 0;
88  if ( rotation(1,1) > rotation(0,0) )
89  i = 1;
90  if ( rotation(2,2) > rotation(i,i) )
91  i = 2;
92  unsigned int j = (i+1)%3;
93  unsigned int k = (j+1)%3;
94  float s = ::IvSqrt( rotation(i,i) - rotation(j,j) - rotation(k,k) + 1.0f );
95  (*this)[i] = 0.5f*s;
96  float recip = 0.5f/s;
97  w = (rotation(k,j) - rotation(j,k))*recip;
98  (*this)[j] = (rotation(j,i) + rotation(i,j))*recip;
99  (*this)[k] = (rotation(k,i) + rotation(i,k))*recip;
100  }
101 
102 } // End of IvQuat::IvQuat()

262 //-------------------------------------------------------------------------------
263 // @ IvQuat::Set()
264 //-------------------------------------------------------------------------------
265 // Set quaternion based on axis-angle
266 //-------------------------------------------------------------------------------
267 void
268 IvQuat::Set( const IvVector3& axis, float angle )
269 {
270  // if axis of rotation is zero vector, just set to identity quat
271  float length = axis.LengthSquared();
272  if ( IvIsZero( length ) )
273  {
274  Identity();
275  return;
276  }
277 
278  // take half-angle
279  angle *= 0.5f;
280 
281  float sintheta, costheta;
282  IvSinCos(angle, sintheta, costheta);
283 
284  float scaleFactor = sintheta/IvSqrt( length );
285 
286  w = costheta;
287  x = scaleFactor * axis.x;
288  y = scaleFactor * axis.y;
289  z = scaleFactor * axis.z;
290 
291 } // End of IvQuat::Set()

294 //-------------------------------------------------------------------------------
295 // @ IvQuat::Set()
296 //-------------------------------------------------------------------------------
297 // Set quaternion based on start and end vectors
298 //
299 // This is a slightly faster method than that presented in the book, and it
300 // doesn't require unit vectors as input. Found on GameDev.net, in an article by
301 // minorlogic. Original source unknown.
302 //-------------------------------------------------------------------------------
303 void
304 IvQuat::Set( const IvVector3& from, const IvVector3& to )
305 {
306  // get axis of rotation
307  IvVector3 axis = from.Cross( to );
308 
309  // get scaled cos of angle between vectors and set initial quaternion
310  Set( from.Dot( to ), axis.x, axis.y, axis.z );
311  // quaternion at this point is ||from||*||to||*( cos(theta), r*sin(theta) )
312 
313  // normalize to remove ||from||*||to|| factor
314  Normalize();
315  // quaternion at this point is ( cos(theta), r*sin(theta) )
316  // what we want is ( cos(theta/2), r*sin(theta/2) )
317 
318  // set up for half angle calculation
319  w += 1.0f;
320 
321  // now when we normalize, we'll be dividing by sqrt(2*(1+cos(theta))), which is
322  // what we want for r*sin(theta) to give us r*sin(theta/2) (see pages 487-488)
323  //
324  // w will become
325  // 1+cos(theta)
326  // ----------------------
327  // sqrt(2*(1+cos(theta)))
328  // which simplifies to
329  // cos(theta/2)
330 
331  // before we normalize, check if vectors are opposing
332  if ( w <= kEpsilon )
333  {
334  // rotate pi radians around orthogonal vector
335  // take cross product with x axis
336  if ( from.z*from.z > from.x*from.x )
337  Set( 0.0f, 0.0f, from.z, -from.y );
338  // or take cross product with z axis
339  else
340  Set( 0.0f, from.y, -from.x, 0.0f );
341  }
342 
343  // normalize again to get rotation quaternion
344  Normalize();
345 
346 } // End of IvQuat::Set()

349 //-------------------------------------------------------------------------------
350 // @ IvQuat::Set()
351 //-------------------------------------------------------------------------------
352 // Set quaternion based on fixed angles
353 //-------------------------------------------------------------------------------
354 void
355 IvQuat::Set( float zRotation, float yRotation, float xRotation )
356 {
357  zRotation *= 0.5f;
358  yRotation *= 0.5f;
359  xRotation *= 0.5f;
360 
361  // get sines and cosines of half angles
362  float Cx, Sx;
363  IvSinCos(xRotation, Sx, Cx);
364 
365  float Cy, Sy;
366  IvSinCos(yRotation, Sy, Cy);
367 
368  float Cz, Sz;
369  IvSinCos(zRotation, Sz, Cz);
370 
371  // multiply it out
372  w = Cx*Cy*Cz - Sx*Sy*Sz;
373  x = Sx*Cy*Cz + Cx*Sy*Sz;
374  y = Cx*Sy*Cz - Sx*Cy*Sz;
375  z = Cx*Cy*Sz + Sx*Sy*Cx;
376 
377 } // End of IvQuat::Set()

487 //-------------------------------------------------------------------------------
488 // Compute quaternion inverse
489 //-------------------------------------------------------------------------------
490 IvQuat
491 Inverse( const IvQuat& quat )
492 {
493  float norm = quat.w*quat.w + quat.x*quat.x + quat.y*quat.y + quat.z*quat.z;
494  // if we're the zero quaternion, just return identity
495  if ( !IvIsZero( norm ) )
496  {
497  ASSERT( false );
498  return IvQuat();
499  }
500 
501  float normRecip = 1.0f / norm;
502  return IvQuat( normRecip*quat.w, -normRecip*quat.x, -normRecip*quat.y,
503  -normRecip*quat.z );
504 
505 } // End of Inverse()

721 //-------------------------------------------------------------------------------
722 // @ Lerp()
723 //-------------------------------------------------------------------------------
724 // Linearly interpolate two quaternions
725 // This will always take the shorter path between them
726 //-------------------------------------------------------------------------------
727 void
728 Lerp( IvQuat& result, const IvQuat& start, const IvQuat& end, float t )
729 {
730  // get cos of "angle" between quaternions
731  float cosTheta = start.Dot( end );
732 
733  // initialize result
734  result = t*end;
735 
736  // if "angle" between quaternions is less than 90 degrees
737  if ( cosTheta >= kEpsilon )
738  {
739  // use standard interpolation
740  result += (1.0f-t)*start;
741  }
742  else
743  {
744  // otherwise, take the shorter path
745  result += (t-1.0f)*start;
746  }
747 
748 } // End of Lerp()

751 //-------------------------------------------------------------------------------
752 // @ Slerp()
753 //-------------------------------------------------------------------------------
754 // Spherical linearly interpolate two quaternions
755 // This will always take the shorter path between them
756 //-------------------------------------------------------------------------------
757 void
758 Slerp( IvQuat& result, const IvQuat& start, const IvQuat& end, float t )
759 {
760  // get cosine of "angle" between quaternions
761  float cosTheta = start.Dot( end );
762  float startInterp, endInterp;
763 
764  // if "angle" between quaternions is less than 90 degrees
765  if ( cosTheta >= kEpsilon )
766  {
767  // if angle is greater than zero
768  if ( (1.0f - cosTheta) > kEpsilon )
769  {
770  // use standard slerp
771  float theta = acosf( cosTheta );
772  float recipSinTheta = 1.0f/IvSin( theta );
773 
774  startInterp = IvSin( (1.0f - t)*theta )*recipSinTheta;
775  endInterp = IvSin( t*theta )*recipSinTheta;
776  }
777  // angle is close to zero
778  else
779  {
780  // use linear interpolation
781  startInterp = 1.0f - t;
782  endInterp = t;
783  }
784  }
785  // otherwise, take the shorter route
786  else
787  {
788  // if angle is less than 180 degrees
789  if ( (1.0f + cosTheta) > kEpsilon )
790  {
791  // use slerp w/negation of start quaternion
792  float theta = acosf( -cosTheta );
793  float recipSinTheta = 1.0f/IvSin( theta );
794 
795  startInterp = IvSin( (t-1.0f)*theta )*recipSinTheta;
796  endInterp = IvSin( t*theta )*recipSinTheta;
797  }
798  // angle is close to 180 degrees
799  else
800  {
801  // use lerp w/negation of start quaternion
802  startInterp = t - 1.0f;
803  endInterp = t;
804  }
805  }
806 
807  result = startInterp*start + endInterp*end;
808 
809 } // End of Slerp()

813 // @ ApproxSlerp()
814 //-------------------------------------------------------------------------------
815 // Approximate spherical linear interpolation of two quaternions
816 // Based on "Hacking Quaternions", Jonathan Blow, Game Developer, March 2002.
817 // See Game Developer, February 2004 for an alternate method.
818 //-------------------------------------------------------------------------------
819 void
820 ApproxSlerp( IvQuat& result, const IvQuat& start, const IvQuat& end, float t )
821 {
822  float cosTheta = start.Dot( end );
823 
824  // correct time by using cosine of angle between quaternions
825  float factor = 1.0f - 0.7878088f*cosTheta;
826  float k = 0.5069269f;
827  factor *= factor;
828  k *= factor;
829 
830  float b = 2*k;
831  float c = -3*k;
832  float d = 1 + k;
833 
834  t = t*(b*t + c) + d;
835 
836  // initialize result
837  result = t*end;
838 
839  // if "angle" between quaternions is less than 90 degrees
840  if ( cosTheta >= kEpsilon )
841  {
842  // use standard interpolation
843  result += (1.0f-t)*start;
844  }
845  else
846  {
847  // otherwise, take the shorter path
848  result += (t-1.0f)*start;
849  }
850 
851 } // End of ApproxSlerp()
