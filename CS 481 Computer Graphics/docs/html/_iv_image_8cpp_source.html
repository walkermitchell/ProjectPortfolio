43 //-------------------------------------------------------------------------------
44 // @ IvReadBinary (unsigned short)
45 //-------------------------------------------------------------------------------
46 // Reads a byte from a file, handling the endian issues
47 //-------------------------------------------------------------------------------
48 static bool IvReadBinary(FILE* file, unsigned short& res)
49 {
50  unsigned char temp[2];
51  if (fread(temp, 1, 2, file) != 2)
52  return false;
53 
54  // swap for endian differences - this assumes that the file is little-endian
55  // but works for both cases of loading machine
56  res = ((unsigned short)temp[0]) | (((unsigned short)temp[1]) << 8);
57  return true;
58 } // End of IvReadBinary

160 // @ IvImage::CreateFromFile
161 //-------------------------------------------------------------------------------
162 // Creates and returns an image representing the given file, or nullptr if the file
163 // does not exist or is an unsupported format
164 //-------------------------------------------------------------------------------
165 IvImage* IvImage::CreateFromFile(const char* pcFilename)
166 {
167  FILE* file = fopen(pcFilename, "rb");
168 
169  if (!file)
170  return 0;
171 
172  unsigned char IDLength;
173  if (!IvReadBinary(file, IDLength))
174  {
175  fclose(file);
176  return 0;
177  }
178 
179  unsigned char colorMapType;
180  if (!IvReadBinary(file, colorMapType))
181  {
182  fclose(file);
183  return 0;
184  }
185 
186  unsigned char imageType;
187  if (!IvReadBinary(file, imageType))
188  {
189  fclose(file);
190  return 0;
191  }
192 
193  // skip 5 bytes of palette info
194  fseek(file, 5, SEEK_CUR);
195 
196  // only care that this is zero
197  unsigned short xOrigin;
198  if (!IvReadBinary(file, xOrigin))
199  {
200  fclose(file);
201  return 0;
202  }
203 
204  // only care that this is zero
205  unsigned short yOrigin;
206  if (!IvReadBinary(file, yOrigin))
207  {
208  fclose(file);
209  return 0;
210  }
211 
212  // MUST byte swap these as needed for the platform
213  unsigned short shortWidth;
214  if (!IvReadBinary(file, shortWidth))
215  {
216  fclose(file);
217  return 0;
218  }
219 
220  unsigned short shortHeight;
221  if (!IvReadBinary(file, shortHeight))
222  {
223  fclose(file);
224  return 0;
225  }
226 
227  unsigned char pixelSize;
228  if (!IvReadBinary(file, pixelSize))
229  {
230  fclose(file);
231  return 0;
232  }
233 
234  unsigned char imageAttributes;
235  if (!IvReadBinary(file, imageAttributes))
236  {
237  fclose(file);
238  return 0;
239  }
240 
241  // skip ID field
242  if (IDLength != 0)
243  fseek(file, IDLength, SEEK_CUR);
244 
245  if (imageType != 2)
246  {
247  fclose(file);
248  return 0;
249  }
250 
251  unsigned int bytesPerPixel = pixelSize >> 3;
252  unsigned int width = shortWidth;
253  unsigned int height = shortHeight;
254 
255  IvImage* image = 0;
256 
257  if (pixelSize == 32)
258  {
259  // RGBA
260  image = new IvImage(width, height, 4, 1);
261  }
262  else if (pixelSize == 24)
263  {
264  // RGB
265  image = new IvImage(width, height, 3, 1);
266  }
267  else
268  {
269  fclose(file);
270  return 0;
271  }
272 
273  unsigned char* pixels = image->GetPixels();
274 
275  unsigned int rowSize = bytesPerPixel * width;
276  int rowStep = (int)rowSize;
277 
278  // Do not load right-to-left images
279  if (imageAttributes & 0x10)
280  {
281  fclose(file);
282  delete image;
283  return 0;
284  }
285 
286  // Must flip top-to-bottom images
287  if (imageAttributes & 0x20)
288  {
289  pixels += rowSize * (height - 1);
290  rowStep = -rowStep;
291  }
292 
293  for (unsigned int row = 0; row < height; row++)
294  {
295  // Load a row of raw pixels into the image directly
296  // from the image file
297  fread(pixels, rowSize, 1, file);
298 
299  unsigned char* tempRow = pixels;
300 
301  // Pixels are read in as BGR(A), and we transpose to RGB(A)
302  for (unsigned int i = 0; i < width; i++)
303  {
304  unsigned char temp = tempRow[2];
305  tempRow[2] = tempRow[0];
306  tempRow[0] = temp;
307 
308  tempRow += bytesPerPixel;
309  }
310 
311  pixels += rowStep;
312  }
313 
314  return image;
315 } // End of IvImage::CreateFromFile
