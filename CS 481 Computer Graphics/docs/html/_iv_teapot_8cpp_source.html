134 //-------------------------------------------------------------------------------
135 // @ BuildMesh()
136 //-------------------------------------------------------------------------------
137 // This routine tessellates a 4x4 Bezier patch into a mesh, and stores it into the
138 // given dynamic arrays. The idea is that we will use the four sets of control
139 // points in the 'v' direction to generate four Bezier curves. We will then
140 // tessellate along those curves to get the control points in the 'u' direction,
141 // which we will use to tessellate our final positions.
142 //
143 // Note that we also need to generate tangents in the 'u' and 'v' directions. The
144 // 'u' direction tangents we can get from our 'u' tessellation. The 'v' direction
145 // tangents we generate by creating a new set of Bezier control points.
146 //-------------------------------------------------------------------------------
147 static void
148 BuildMesh( const IvVector3 controls[4][4], std::vector<IvTNPVertex>& vertices, std::vector<unsigned int>& indices )
149 {
150  const unsigned int grid = 7;
151  const float h = 1.0f/(float)(grid-1);
152  const float h2 = h*h;
153  const float h3 = h2*h;
154 
155  unsigned int indexDiff = vertices.size();
156 
157  // set up forward differencing parameters for curves in v-direction
158  IvVector3 v[4]; // position controls
159  IvVector3 dv1[4];
160  IvVector3 dv2[4];
161  IvVector3 dv3[4];
162  IvVector3 vtan[4]; // velocity controls (for v-direction)
163  IvVector3 dvtan1[4];
164  IvVector3 dvtan2[4];
165  for ( unsigned int k = 0; k < 4; ++k )
166  {
167  IvVector3 A = controls[3][k]
168  - 3.0f*controls[2][k]
169  + 3.0f*controls[1][k]
170  - controls[0][k];
171  IvVector3 B = 3.0f*controls[2][k]
172  - 6.0f*controls[1][k]
173  + 3.0f*controls[0][k];
174  IvVector3 C = 3.0f*controls[1][k]
175  - 3.0f*controls[0][k];
176  // scale down by 3/4
177  A *= 0.75f;
178  B *= 0.75f;
179  C *= 0.75f;
180  dv1[k] = h3*A + h2*B + h*C;
181  dv3[k] = 6.0f*h3*A;
182  dv2[k] = dv3[k] + 2.0f*h2*B;
183  v[k] = 0.75f*controls[0][k];
184  dvtan1[k] = 3.0f*h2*A + 2.0f*h*B;
185  dvtan2[k] = 6.0f*h2*A;
186  vtan[k] = C;
187  }
188 
189  // start tessellating in the 'v' direction
190  for ( unsigned int j = 0; j < grid; ++j )
191  {
192  // now build forward differencing variables for u-direction curve
193  // using control points we've just tesselated
194  IvVector3 A = v[3]
195  - 3.0f*v[2]
196  + 3.0f*v[1]
197  - v[0];
198  IvVector3 B = 3.0f*v[2]
199  - 6.0f*v[1]
200  + 3.0f*v[0];
201  IvVector3 C = 3.0f*v[1]
202  - 3.0f*v[0];
203  IvVector3 dP1 = h3*A + h2*B + h*C;
204  IvVector3 dP3 = 6.0f*h3*A;
205  IvVector3 dP2 = dP3 + 2.0f*h2*B;
206  IvVector3 P = v[0];
207  IvVector3 ddu1 = 3.0f*h2*A + 2.0f*h*B;
208  IvVector3 ddu2 = 6.0f*h2*A;
209  IvVector3 du = C;
210 
211  // also need to forward difference v-velocities
212  A = vtan[3]
213  - 3.0f*vtan[2]
214  + 3.0f*vtan[1]
215  - vtan[0];
216  B = 3.0f*vtan[2]
217  - 6.0f*vtan[1]
218  + 3.0f*vtan[0];
219  C = 3.0f*vtan[1]
220  - 3.0f*vtan[0];
221  IvVector3 ddv1 = h3*A + h2*B + h*C;
222  IvVector3 ddv3 = 6.0f*h3*A;
223  IvVector3 ddv2 = ddv3 + 2.0f*h2*B;
224  IvVector3 dv = vtan[0];
225 
226  // tessellate in the 'u' direction
227  for ( unsigned int i = 0; i < grid; ++i )
228  {
229  IvTNPVertex vertex;
230  vertex.position = P;
231  IvVector2 coord( float(i)/float(grid-1), float(j)/float(grid-1) );
232  vertex.texturecoord = coord;
233  vertex.normal = Cross( dv, du );
234  vertex.normal.Normalize();
235  // whoops, singularity
236  if ( vertex.normal.IsZero() )
237  {
238  // this is not a general solution, but it works for the teapot
239  // take average of face normals using previous row of vertices (if there is one)
240  if ( j > 0 )
241  {
242  vertex.normal.Zero();
243  for ( unsigned int k = 0; k < grid-1; ++k )
244  {
245  unsigned int index0 = grid*(j-1) + k + indexDiff;
246  unsigned int index1 = grid*(j-1) + k+1 + indexDiff;
247  IvVector3 faceNormal = Cross(vertices[index0].position - P,
248  vertices[index1].position - P);
249  faceNormal.Normalize();
250  vertex.normal += faceNormal;
251  }
252  vertex.normal.Normalize();
253  }
254  }
255 
256  vertices.push_back( vertex );
257 
258  if ( i > 0 && j > 0 )
259  {
260  unsigned int index0 = grid*(j-1) + (i-1);
261  unsigned int index1 = grid*(j-1) + i;
262  unsigned int index2 = grid*j + (i-1);
263  unsigned int index3 = grid*j + i;
264  indices.push_back( index0 + indexDiff );
265  indices.push_back( index2 + indexDiff );
266  indices.push_back( index3 + indexDiff );
267  indices.push_back( index0 + indexDiff );
268  indices.push_back( index3 + indexDiff );
269  indices.push_back( index1 + indexDiff );
270  }
271 
272  P += dP1;
273  dP1 += dP2;
274  dP2 += dP3;
275 
276  du += ddu1;
277  ddu1 += ddu2;
278 
279  dv += ddv1;
280  ddv1 += ddv2;
281  ddv2 += ddv3;
282  }
283 
284  // increment v-direction curves to get new control points
285  for ( unsigned int k = 0; k < 4; ++k )
286  {
287  v[k] += dv1[k];
288  dv1[k] += dv2[k];
289  dv2[k] += dv3[k];
290  vtan[k] += dvtan1[k];
291  dvtan1[k] += dvtan2[k];
292  }
293  }
294 }

296 //-------------------------------------------------------------------------------
297 // @ BuildTeapot()
298 //-------------------------------------------------------------------------------
299 // This routine builds all the mesh data for the teapot, using the BuildMesh()
300 // routine above. It cycles through all the patch data and reflects it to
301 // get a full mesh.
302 //-------------------------------------------------------------------------------
303 void
304 BuildTeapot( std::vector<IvTNPVertex>& vertices, std::vector<unsigned int>& indices )
305 {
306  IvVector3 p[4][4], q[4][4], r[4][4], s[4][4];
307 
308  for (unsigned int i = 0; i < 10; i++)
309  {
310  for (unsigned int j = 0; j < 4; j++)
311  {
312  for (unsigned int k = 0; k < 4; k++) {
313  for (unsigned int l = 0; l < 3; l++)
314  {
315  p[3-j][k][l] = cpdata[patchdata[i][j * 4 + k]][l];
316  q[3-j][k][l] = cpdata[patchdata[i][j * 4 + (3 - k)]][l];
317  if (l == 1)
318  q[3-j][k][l] *= -1.0;
319  if (i < 6)
320  {
321  r[3-j][k][l] =
322  cpdata[patchdata[i][j * 4 + (3 - k)]][l];
323  if (l == 0)
324  r[3-j][k][l] *= -1.0;
325  s[3-j][k][l] = cpdata[patchdata[i][j * 4 + k]][l];
326  if (l == 0)
327  s[3-j][k][l] *= -1.0;
328  if (l == 1)
329  s[3-j][k][l] *= -1.0;
330  }
331  }
332  }
333  }
334 
335  BuildMesh(p, vertices, indices);
336  BuildMesh(q, vertices, indices);
337  if ( i < 6 )
338  {
339  BuildMesh(r, vertices, indices);
340  BuildMesh(s, vertices, indices);
341  }
342  }
