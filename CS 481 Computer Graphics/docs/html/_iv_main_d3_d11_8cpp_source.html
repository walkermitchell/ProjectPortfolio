221 bool InitDevice(unsigned int width, unsigned int height, bool fullscreen, bool vsync)
222 {
223  int numerator, denominator;
224  if (vsync)

226  if (!GetRefreshRate(width, height, numerator, denominator))

234  numerator = 0;

238 
239  // Set up the swap chain description
240  DXGI_SWAP_CHAIN_DESC swapChainDesc;
241  ZeroMemory(&swapChainDesc, sizeof(swapChainDesc));
242  swapChainDesc.BufferCount = 1; // one backbuffer
243  swapChainDesc.BufferDesc.Width = width;
244  swapChainDesc.BufferDesc.Height = height;
245  swapChainDesc.BufferDesc.Format = DXGI_FORMAT_R8G8B8A8_UNORM_SRGB;
246  swapChainDesc.BufferDesc.RefreshRate.Numerator = numerator;
247  swapChainDesc.BufferDesc.RefreshRate.Denominator = denominator;
248  swapChainDesc.BufferUsage = DXGI_USAGE_RENDER_TARGET_OUTPUT;
249  swapChainDesc.OutputWindow = gHwnd;
250  swapChainDesc.SampleDesc.Count = 1;
251  swapChainDesc.SampleDesc.Quality = 0;
252  swapChainDesc.Windowed = !fullscreen;
253 
254 #if _DEBUG

256 #else
257  UINT deviceFlags = 0;
258 #endif
259  // Create the swap chain, Direct3D device, and Direct3D device context.
260  HRESULT result = D3D11CreateDeviceAndSwapChain(nullptr, D3D_DRIVER_TYPE_HARDWARE, nullptr, deviceFlags, nullptr, 0,
261  D3D11_SDK_VERSION, &swapChainDesc, &gSwapChain,
262  &gDevice, nullptr, &gContext);
263  if (FAILED(result))
264  {
265  //*** fallback to WARP?

279 bool GetRefreshRate(unsigned int width, unsigned int height,
280  int& numerator, int& denominator)
281 {
282  IDXGIFactory* factory;
283 
284  HRESULT result = CreateDXGIFactory1(__uuidof(IDXGIFactory1), (void**)&factory);
285  if (FAILED(result))
286  {
287  result = CreateDXGIFactory(__uuidof(IDXGIFactory), (void**)&factory);
288  }
289  if (FAILED(result))
290  {
291  return false;
292  }
293 
294  // We start by getting the primary graphics interface, and from that getting the primary output
295  IDXGIAdapter* adapter;
296  IDXGIOutput* adapterOutput;
297  result = factory->EnumAdapters(0, &adapter);
298  if (SUCCEEDED(result))
299  {
300  result = adapter->EnumOutputs(0, &adapterOutput);
301  }
302  else
303  {
304  adapter->Release();
305  factory->Release();
306  return false;
307  }
308 
309  // Now we enumerate all the display modes that match 32-bit RGBA w/sRGB gamma
310  unsigned int numModes;
311  DXGI_MODE_DESC* displayModeList = nullptr;
312  result = adapterOutput->GetDisplayModeList(DXGI_FORMAT_R8G8B8A8_UNORM_SRGB, DXGI_ENUM_MODES_INTERLACED, &numModes, nullptr);
313  if (SUCCEEDED(result))
314  {
315  displayModeList = new DXGI_MODE_DESC[numModes];
316  if (nullptr != displayModeList)
317  {
318  result = adapterOutput->GetDisplayModeList(DXGI_FORMAT_R8G8B8A8_UNORM_SRGB, DXGI_ENUM_MODES_INTERLACED, &numModes, displayModeList);
319  }
320  }
321  if (nullptr == displayModeList || FAILED(result))
322  {
323  adapterOutput->Release();
324  adapter->Release();
325  factory->Release();
326  return false;
327  }
328 
329  // Now go through all the display modes and find the one that matches the screen width and height.
330  // When a match is found store the numerator and denominator of the refresh rate for that monitor.
331  for (unsigned int i = 0; i<numModes; i++)
332  {
333  if (displayModeList[i].Width == width)
334  {
335  if (displayModeList[i].Height == height)
336  {
337  numerator = displayModeList[i].RefreshRate.Numerator;
338  denominator = displayModeList[i].RefreshRate.Denominator;
339  }
340  }
341  }
342 
343  // Get the adapter (video card) description.
344  //DXGI_ADAPTER_DESC adapterDesc;
345  //result = adapter->GetDesc(&adapterDesc);
346  //if (FAILED(result))
347  //{
348  // return false;
349  //}
350 
352  //m_videoCardMemory = (int)(adapterDesc.DedicatedVideoMemory / 1024 / 1024);
353 
355  //unsigned int stringLength;
356  //error = wcstombs_s(&stringLength, m_videoCardDescription, 128, adapterDesc.Description, 128);
357  //if (error != 0)
358  //{
359  // return false;
360  //}
361 
362  // Clean up
363  delete[] displayModeList;
364  adapterOutput->Release();
365  adapter->Release();
366  factory->Release();

424 LRESULT CALLBACK WndProc(HWND hWnd, UINT message, WPARAM wParam, LPARAM lParam)
425 {
426  PAINTSTRUCT ps;
427  HDC hdc;
428 
429  switch (message)
430  {
431  case WM_PAINT:
432  hdc = BeginPaint(hWnd, &ps);
433  EndPaint(hWnd, &ps);
434  break;
435 
436  case WM_CLOSE:
437  case WM_DESTROY:
438  PostQuitMessage(0);
439  break;
440 
441  case WM_ACTIVATE:
442  if (IvRenderer::mRenderer)
443  {
444  IvRenderer::mRenderer->Activate(wParam != WA_INACTIVE);
445  }
446  break;
447 
448  case WM_SIZE:
449  if (gSwapChain)
450  {
451  int width = LOWORD(lParam);
452  int height = HIWORD(lParam);
453 
454  gContext->OMSetRenderTargets(0, 0, 0);
455 
456  // Release all outstanding references to the swap chain's buffers.
457  if (IvRenderer::mRenderer)
458  {
459  reinterpret_cast<IvRendererD3D11*>(IvRenderer::mRenderer)->ReleaseTargets();
460  }
461  if (gRenderTargetView)
462  {
463  gRenderTargetView->Release();
464  gRenderTargetView = nullptr;
465  }
466  if (gDepthStencilView)
467  {
468  gDepthStencilView->Release();
469  gDepthStencilView = nullptr;
470  }
471  if (gDepthStencilBuffer)
472  {
473  gDepthStencilBuffer->Release();
474  gDepthStencilBuffer = nullptr;
475  }
476 
477  HRESULT hr;
478  // Preserve the existing buffer count and format.
479  // Automatically choose the width and height to match the client rect for HWNDs.
480  hr = gSwapChain->ResizeBuffers(0, width, height, DXGI_FORMAT_UNKNOWN, 0);
481 
482  // Create a render target view
483  ID3D11Texture2D* backBuffer = nullptr;
484  hr = gSwapChain->GetBuffer(0, __uuidof(ID3D11Texture2D), reinterpret_cast<void**>(&backBuffer));
485  if (FAILED(hr))
486  {
487  PostQuitMessage(0);
488  return E_FAIL;
489  }
490  hr = gDevice->CreateRenderTargetView(backBuffer, nullptr, &gRenderTargetView);
491  backBuffer->Release();
492  if (FAILED(hr))
493  {
494  PostQuitMessage(0);
495  return E_FAIL;
496  }
497 
498  // Create depth/stencil buffer
499  D3D11_TEXTURE2D_DESC depthBufferDesc;
500  ZeroMemory(&depthBufferDesc, sizeof(depthBufferDesc));
501  depthBufferDesc.Width = width;
502  depthBufferDesc.Height = height;
503  depthBufferDesc.MipLevels = 1;
504  depthBufferDesc.ArraySize = 1;
505  depthBufferDesc.Format = DXGI_FORMAT_D24_UNORM_S8_UINT;
506  depthBufferDesc.SampleDesc.Count = 1;
507  depthBufferDesc.SampleDesc.Quality = 0;
508  depthBufferDesc.Usage = D3D11_USAGE_DEFAULT;
509  depthBufferDesc.BindFlags = D3D11_BIND_DEPTH_STENCIL;
510  depthBufferDesc.CPUAccessFlags = 0;
511  depthBufferDesc.MiscFlags = 0;
512  hr = gDevice->CreateTexture2D(&depthBufferDesc, nullptr, &gDepthStencilBuffer);
513  if (FAILED(hr))
514  {
515  gRenderTargetView->Release();
516  gRenderTargetView = 0;
517  PostQuitMessage(0);
518  return E_FAIL;
519  }
520 
521  // And now create the depth stencil view
522  D3D11_DEPTH_STENCIL_VIEW_DESC depthStencilViewDesc;
523  ZeroMemory(&depthStencilViewDesc, sizeof(depthStencilViewDesc));
524  depthStencilViewDesc.Format = DXGI_FORMAT_D24_UNORM_S8_UINT;
525  depthStencilViewDesc.ViewDimension = D3D11_DSV_DIMENSION_TEXTURE2D;
526  depthStencilViewDesc.Texture2D.MipSlice = 0;
527  hr = gDevice->CreateDepthStencilView(gDepthStencilBuffer, &depthStencilViewDesc, &gDepthStencilView);
528  if (FAILED(hr))
529  {
530  gDepthStencilBuffer->Release();
531  gDepthStencilBuffer = 0;
532  gRenderTargetView->Release();
533  gRenderTargetView = 0;
534  PostQuitMessage(0);
535  return E_FAIL;
536  }
537 
538  // Bind the render target view and depth stencil buffer to the output render pipeline.
539  gContext->OMSetRenderTargets(1, &gRenderTargetView, gDepthStencilView);
540  }
541 
542  if (IvRenderer::mRenderer)
543  {
544  reinterpret_cast<IvRendererD3D11*>(IvRenderer::mRenderer)->SetTargets(gRenderTargetView, gDepthStencilView);
545  IvRenderer::mRenderer->Resize(LOWORD(lParam), HIWORD(lParam));
546  }
547  break;
548 
549  case WM_KEYDOWN:
550  case WM_SYSKEYDOWN:
551  {
552  DWORD dwMask = (1 << 29);
553  bool bAltDown = ((lParam & dwMask) != 0);
554 
555  OnKeyboard((UINT)wParam, true, bAltDown);
556  }
557  break;
558 
559  case WM_KEYUP:
560  case WM_SYSKEYUP:
561  {
562  DWORD dwMask = (1 << 29);
563  bool bAltDown = ((lParam & dwMask) != 0);
564 
565  OnKeyboard((UINT)wParam, false, bAltDown);
566  }
567  break;
568 
569  case WM_LBUTTONDOWN:
570  case WM_RBUTTONDOWN:
571  case WM_MBUTTONDOWN:
572  case WM_LBUTTONUP:
573  case WM_RBUTTONUP:
574  case WM_MBUTTONUP:
575  {
576  bool bLeftDown = ((wParam & MK_LBUTTON) != 0);
577  bool bRightDown = ((wParam & MK_RBUTTON) != 0);
578  bool bMiddleDown = ((wParam & MK_MBUTTON) != 0);
579 
580  int xPos = GET_X_LPARAM(lParam);
581  int yPos = GET_Y_LPARAM(lParam);
582  if (!gIsFullscreen)
583  {
584  RECT rc = { 0, 0, 640, 480 };
585  AdjustWindowRect(&rc, WS_OVERLAPPEDWINDOW, FALSE);
586  xPos -= rc.left;
587  yPos -= rc.top;
588  }
589 
590  OnMouse(bLeftDown, bRightDown, bMiddleDown, false, false, 0, xPos, yPos);
591  }
592  break;
593 
594  default:
595  return DefWindowProc(hWnd, message, wParam, lParam);
596  }
597 
598  return 0;
599 }

671 //
672 // CommandLineWToArgvA - converts Windows command line string to Unix-style command line arguments
673 //
674 // Based on code by Alexander A. Telyatnikov (http://alter.org.ua/en/docs/win/args/)
675 //
676 PCHAR*
677 CommandLineWToArgvA(PWCHAR CmdLine, int* _argc)
678 {
679  PCHAR* argv;
680  PCHAR _argv;
681  ULONG len;
682  ULONG argc;
683  WCHAR w;
684  ULONG i, j;
685 
686  BOOLEAN in_QM;
687  BOOLEAN in_TEXT;
688  BOOLEAN in_SPACE;
689 
690  len = wcslen(CmdLine);
691  i = ((len + 2) / 2)*sizeof(PVOID)+sizeof(PVOID);
692 
693  argv = (PCHAR*)GlobalAlloc(GMEM_FIXED,
694  i + (len + 2)*sizeof(CHAR));
695 
696  _argv = (PCHAR)(((PCHAR)argv) + i);
697 
698  argc = 0;
699  argv[argc] = _argv;
700  in_QM = FALSE;
701  in_TEXT = FALSE;
702  in_SPACE = TRUE;
703  i = 0;
704  j = 0;
705 
706  while (w = CmdLine[i]) {
707  int a = wctob(w); // convert to char
708  assert(a >= 0);
709  if (in_QM) {
710  if (a == '\"') {
711  in_QM = FALSE;
712  }
713  else {
714  _argv[j] = a;
715  j++;
716  }
717  }
718  else {
719  switch (a) {
720  case '\"':
721  in_QM = TRUE;
722  in_TEXT = TRUE;
723  if (in_SPACE) {
724  argv[argc] = _argv + j;
725  argc++;
726  }
727  in_SPACE = FALSE;
728  break;
729  case ' ':
730  case '\t':
731  case '\n':
732  case '\r':
733  if (in_TEXT) {
734  _argv[j] = '\0';
735  j++;
736  }
737  in_TEXT = FALSE;
738  in_SPACE = TRUE;
739  break;
740  default:
741  in_TEXT = TRUE;
742  if (in_SPACE) {
743  argv[argc] = _argv + j;
