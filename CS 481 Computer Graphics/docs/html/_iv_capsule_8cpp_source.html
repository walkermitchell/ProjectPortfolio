142 // @ IvCapsule::Set()
143 //-------------------------------------------------------------------------------
144 // Set capsule based on set of points
145 //-------------------------------------------------------------------------------
146 void
147 IvCapsule::Set( const IvVector3* points, unsigned int nPoints )
148 {
149  ASSERT( points );
150  IvVector3 centroid;
151 
152  // compute covariance matrix
153  IvMatrix33 C;
154  IvComputeCovarianceMatrix( C, centroid, points, nPoints );
155 
156  // get main axis
157  IvVector3 w, u, v;
158  IvGetRealSymmetricEigenvectors( w, u, v, C );
159 
160  float maxDistSq = 0.0f;
161 
162  // for each point do
163  unsigned int i;
164  for ( i = 0; i < nPoints; ++i )
165  {
166  // compute radius
167  IvVector3 diff = points[i]-centroid;
168  float proj = diff.Dot(w);
169 
170  float distSq = diff.Dot(diff) - proj*proj;
171  if (distSq > maxDistSq)
172  maxDistSq = distSq;
173  }
174 
175  mRadius = ::IvSqrt(maxDistSq);
176 
177  // now set endcaps
178  // for each point do
179  float t0 = FLT_MAX;
180  float t1 = FLT_MIN;
181  for ( i = 0; i < nPoints; ++i )
182  {
183  IvVector3 localTrans = points[i]-centroid;
184  float uCoord = localTrans.Dot(u);
185  float vCoord = localTrans.Dot(v);
186  float wCoord = localTrans.Dot(w);
187 
188  float radical = maxDistSq - uCoord*uCoord - vCoord*vCoord;
189  if ( radical > kEpsilon )
190  radical = ::IvSqrt( radical );
191  else
192  radical = 0.0f;
193 
194  float test = wCoord + radical;
195  if ( test < t0 )
196  t0 = test;
197 
198  test = wCoord - radical;
199  if ( test > t1 )
200  t1 = test;
201  }
202 
203  // set up segment
204  if ( t0 < t1 )
205  {
206  mSegment.Set( centroid + t0*w, centroid + t1*w );
207  }
208  else
209  {
210  // is sphere
211  IvVector3 center = centroid + 0.5f*(t0+t1)*w;
212  mSegment.Set( center, center );
213  }
214 
215 } // End of IvCapsule::Set()

309 //----------------------------------------------------------------------------
310 // @ IvCapsule::Classify()
311 // ---------------------------------------------------------------------------
312 // Return signed distance between capsule and plane
313 //-----------------------------------------------------------------------------
314 float
315 IvCapsule::Classify( const IvPlane& plane ) const
316 {
317  float s0 = plane.Test( mSegment.GetEndpoint0() );
318  float s1 = plane.Test( mSegment.GetEndpoint1() );
319 
320  // points on opposite sides or intersecting plane
321  if ( s0*s1 <= 0.0f )
322  return 0.0f;
323 
324  // intersect if either endpoint is within radius distance of plane
325  if( ::IvAbs(s0) <= mRadius || ::IvAbs(s1) <= mRadius )
326  return 0.0f;
327 
328  // return signed distance
329  return ( ::IvAbs(s0) < ::IvAbs(s1) ? s0 : s1 );
330 }

333 //----------------------------------------------------------------------------
334 // @ ::Merge()
335 // ---------------------------------------------------------------------------
336 // Merge two capsules together to create a new one
337 // See _3D Game Engine Design_ (Eberly) for more details
338 //-----------------------------------------------------------------------------
339 void
340 Merge( IvCapsule& result, const IvCapsule& b0, const IvCapsule& b1 )
341 {
342  // 1) get line through center
343 
344  // origin is average of centers
345  IvVector3 origin = 0.5f*(b0.mSegment.GetCenter() + b1.mSegment.GetCenter());
346 
347  // direction is average of directions
348  IvVector3 direction0 = b0.mSegment.GetDirection();
349  direction0.Normalize();
350  IvVector3 direction1 = b1.mSegment.GetDirection();
351  direction1.Normalize();
352  if (direction0.Dot(direction1) < 0.0f)
353  direction1 = -direction1;
354  IvVector3 direction = direction0 + direction1;
355 
356  // create line
357  IvLine3 line( origin, direction );
358 
359  // 2) compute radius
360 
361  // radius will be the maximum of the distance to each endpoint,
362  // plus the corresponding capsule radius
363  float t;
364  float radius = Distance( line, b0.mSegment.GetEndpoint0(), t ) + b0.mRadius;
365  float tempRadius = Distance( line, b0.mSegment.GetEndpoint1(), t ) + b0.mRadius;
366  if ( tempRadius > radius )
367  radius = tempRadius;
368  tempRadius = Distance( line, b1.mSegment.GetEndpoint0(), t ) + b1.mRadius;
369  if ( tempRadius > radius )
370  radius = tempRadius;
371  tempRadius = Distance( line, b1.mSegment.GetEndpoint1(), t ) + b1.mRadius;
372  if ( tempRadius > radius )
373  radius = tempRadius;
374 
375  // 3) compute parameters of endpoints on line
376 
377  // we compute a sphere at each original endpoint, and set the endcaps of the
378  // new capsule around them, minimizing size
379 
380  // capsule 0, endpoint 0
381  float radiusDiffSq = radius - b0.GetRadius();
382  radiusDiffSq *= radiusDiffSq;
383  IvVector3 originDiff = line.GetOrigin() - b0.mSegment.GetEndpoint0();
384  // compute coefficients for quadratic
385  float halfb = line.GetDirection().Dot( originDiff );
386  float c = originDiff.Dot(originDiff) - radiusDiffSq;
387  // this is sqrt( b^2 - 4ac )/2, but a lot of terms have cancelled out
388  float radical = halfb*halfb - c;
389  if ( radical > kEpsilon )
390  radical = ::IvSqrt( radical );
391  else
392  radical = 0.0f;
393  // and this is adding -b/2 +/- the above
394  float t0 = -halfb - radical;
395  float t1 = -halfb + radical;
396 
397  // capsule 0, endpoint 1
398  originDiff = line.GetOrigin() - b0.mSegment.GetEndpoint1();
399  halfb = line.GetDirection().Dot( originDiff );
400  c = originDiff.Dot(originDiff) - radiusDiffSq;
401  radical = halfb*halfb - c;
402  if ( radical > kEpsilon )
403  radical = ::IvSqrt( radical );
404  else
405  radical = 0.0f;
406  float param = -halfb - radical;
407  if ( param < t0 )
408  t0 = param;
409  param = -halfb + radical;
410  if ( param > t1 )
411  t1 = param;
412 
413  // capsule 1, endpoint 0
414  radiusDiffSq = radius - b1.GetRadius();
415  radiusDiffSq *= radiusDiffSq;
416  originDiff = line.GetOrigin() - b1.mSegment.GetEndpoint0();
417  halfb = line.GetDirection().Dot( originDiff );
418  c = originDiff.Dot(originDiff) - radiusDiffSq;
419  radical = halfb*halfb - c;
420  if ( radical > kEpsilon )
421  radical = ::IvSqrt( radical );
422  else
423  radical = 0.0f;
424  param = -halfb - radical;
425  if ( param < t0 )
426  t0 = param;
427  param = -halfb + radical;
428  if ( param > t1 )
429  t1 = param;
430 
431  // capsule 1, endpoint 1
432  originDiff = line.GetOrigin() - b1.mSegment.GetEndpoint1();
433  halfb = line.GetDirection().Dot( originDiff );
434  c = originDiff.Dot(originDiff) - radiusDiffSq;
435  radical = halfb*halfb - c;
436  if ( radical > kEpsilon )
437  radical = ::IvSqrt( radical );
438  else
439  radical = 0.0f;
440  param = -halfb - radical;
441  if ( param < t0 )
442  t0 = param;
443  param = -halfb + radical;
444  if ( param > t1 )
445  t1 = param;
446 
447  // set capsule
448  result.mRadius = radius;
449  if ( t0 < t1 )
450  {
451  result.mSegment.Set( line.GetOrigin() + t0*line.GetDirection(),
452  line.GetOrigin() + t1*line.GetDirection() );
453  }
454  else
455  {
456  // is sphere
457  IvVector3 center = line.GetOrigin() + 0.5f*(t0+t1)*line.GetDirection();
458  result.mSegment.Set( center, center );
459  }
460 
461 } // End of ::Merge()

465 //----------------------------------------------------------------------------
466 // @ IvCapsule::ComputeCollision()
467 // ---------------------------------------------------------------------------
468 // Compute parameters for collision between capsule and capsule
469 //-----------------------------------------------------------------------------
470 bool
471 IvCapsule::ComputeCollision( const IvCapsule& other, IvVector3& collisionNormal,
472  IvVector3& collisionPoint, float& penetration ) const
473 {
474  float radiusSum = mRadius + other.mRadius;
475 
476  // if colliding
477  float s, t;
478  float distancesq = DistanceSquared( mSegment, other.mSegment, s, t );
479 
480  if ( distancesq <= radiusSum*radiusSum )
481  {
482  // calculate our values
483  IvVector3 nearPoint0 = (1.0f-s)*mSegment.GetEndpoint0() + s*mSegment.GetEndpoint1();
484  IvVector3 nearPoint1 = (1.0f-t)*other.mSegment.GetEndpoint0() + t*other.mSegment.GetEndpoint1();
485 
486  collisionNormal = nearPoint1 - nearPoint0;
487 
488  // penetration is distance - radii
489  float distance = ::IvSqrt(distancesq);
490  penetration = radiusSum - distance;
491  collisionNormal.Normalize();
492 
493  // collision point is average of penetration
494  // weighted towards lighter object
495  float t = 0.5f;//mMass/(mMass + other->mMass);
496  collisionPoint = (1.0f-t)*(nearPoint0 + mRadius*collisionNormal)
497  + t*(nearPoint1 - other.mRadius*collisionNormal);
498 
499  return true;
500  }
501 
502  return false;
503 }
