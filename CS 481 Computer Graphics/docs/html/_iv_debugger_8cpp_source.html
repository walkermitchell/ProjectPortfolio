87 // @ AmIBeingDebugged()
88 //-------------------------------------------------------------------------------
89 // From Mac Developer Library
90 //-------------------------------------------------------------------------------
91 static bool AmIBeingDebugged(void)
92 // Returns true if the current process is being debugged (either
93 // running under the debugger or has a debugger attached post facto).
94 {
95  int junk;
96  int mib[4];
97  struct kinfo_proc info;
98  size_t size;
99 
100  // Initialize the flags so that, if sysctl fails for some bizarre
101  // reason, we get a predictable result.
102 
103  info.kp_proc.p_flag = 0;
104 
105  // Initialize mib, which tells sysctl the info we want, in this case
106  // we're looking for information about a specific process ID.
107 
108  mib[0] = CTL_KERN;
109  mib[1] = KERN_PROC;
110  mib[2] = KERN_PROC_PID;
111  mib[3] = getpid();
112 
113  // Call sysctl.
114 
115  size = sizeof(info);
116  junk = sysctl(mib, sizeof(mib) / sizeof(*mib), &info, &size, nullptr, 0);
117  assert(junk == 0);
118 
119  // We're being debugged if the P_TRACED flag is set.
120 
121  return ( (info.kp_proc.p_flag & P_TRACED) != 0 );
122 }
