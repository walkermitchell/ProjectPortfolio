60 IvMersenne::IvMersenne( UInt32 seed )
61 {
62  mStateVector = new UInt32[kN];
63  // initialize the state vector
64  mStateVector[0] = seed;
65  for ( UInt32 i = 1; i < kN; ++i )
66  {
67  // randomly generate the initial state values
68  // this generator ensures that both the most significant bits and least
69  // significant bits are well mixed
70  mStateVector[i] = (1812433253UL * (mStateVector[i-1] ^ (mStateVector[i-1] >> 30)) + i);
71  }
72  mCurrentEntry = kN;
73 }

90 UInt32
91 IvMersenne::Random()
92 {
93  // if we've exhausted the current state array
94  if (mCurrentEntry >= kN)
95  {
96  // regenerate
97  // do the first N-M values
98  for ( UInt32 k = 0; k < kN-kM; ++k )
99  {
100  UInt32 temp = ((mStateVector[k]&kUpperMask) | (mStateVector[k+1]&kLowerMask));
101  // new state value is: (value from M away) XOR (new blended value shifted) XOR (twist)
102  mStateVector[k] = mStateVector[k + kM] ^ (temp >> 1) ^ kMag01[temp & kTwistMask];
103  }
104  // then all but the last value, being careful of wrapping
105  for ( UInt32 k = kN-kM; k < kN-1; ++k )
106  {
107  UInt32 temp = ((mStateVector[k]&kUpperMask) | (mStateVector[k+1]&kLowerMask));
108  mStateVector[k] = mStateVector[k + (kM-kN)] ^ (temp >> 1) ^ kMag01[temp & kTwistMask];
109  }
110  // then the last
111  UInt32 temp = ((mStateVector[kN-1]&kUpperMask) | (mStateVector[0]&kLowerMask));
112  mStateVector[kN-1] = mStateVector[kM-1] ^ (temp >> 1) ^ kMag01[temp & kTwistMask];
113 
114  mCurrentEntry = 0;
115  }
116 
117  // get the next value
118  UInt32 y = mStateVector[mCurrentEntry++];
119 
120  // temper it
121  y ^= y >> kU;
122  y ^= (y << kS) & kB;
123  y ^= (y << kT) & kC;
124  y ^= y >> kL;
125 
126  return y;
127 }
