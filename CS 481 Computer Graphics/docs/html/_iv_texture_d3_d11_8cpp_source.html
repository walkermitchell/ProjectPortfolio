49 // @ Convert24Bit()
50 //-------------------------------------------------------------------------------
51 // Utility function to convert 24-bit color data to 32-bit
52 //-------------------------------------------------------------------------------
53 static void Convert24Bit(void* outData, void* inData, unsigned int width, unsigned int height)
54 {
55  unsigned char* inComponent = (unsigned char*)inData;
56  unsigned char* outComponent = (unsigned char*)outData;
57  for (unsigned int j = 0; j < height; ++j)
58  {
59  for (unsigned int i = 0; i < width; ++i)
60  {
61  // RGB
62  *outComponent++ = *inComponent++;
63  *outComponent++ = *inComponent++;
64  *outComponent++ = *inComponent++;
65  // A
66  *outComponent++ = 0xff;
67  }
68  }
69 }

72 // @ IvTextureD3D11::Create()

75 //-------------------------------------------------------------------------------
76 bool
77 IvTextureD3D11::Create(unsigned int width, unsigned int height, IvTextureFormat format,
78  void* data, IvDataUsage usage, ID3D11Device* device)
79 {
80  if (width == 0 || height == 0 || mTexturePtr)
81  {
82  return false;
83  }
84 
85  if (usage == kImmutableUsage && !data)
86  {
87  return false;
88  }
89 
90  mWidth = width;
91  mHeight = height;
92  mFormat = format;
93  mUsage = usage;
94 
95  unsigned int texelSize = sInternalTextureFormatSize[mFormat];
96  unsigned int externalTexelSize = sExternalTextureFormatSize[mFormat];
97 
98  mLevelCount = 1;
99 
100  // allocate level for BeginLoadData/EndLoadData
101  if (usage != kImmutableUsage)
102  {
103  mLevels = new Level[mLevelCount];
104 
105  mLevels[0].mData = nullptr;
106  mLevels[0].mWidth = mWidth;
107  mLevels[0].mHeight = mHeight;
108  mLevels[0].mSize = mWidth*mHeight*texelSize;
109  }
110 
111  D3D11_SUBRESOURCE_DATA* subresourcePtr = nullptr;
112  D3D11_SUBRESOURCE_DATA subresourceData;
113  void* pixelData = data;
114  if (data)
115  {
116  // if 24-bit color, we need to convert to 32-bit color
117  if (data && kRGB24TexFmt == format)
118  {
119  pixelData = new unsigned char[texelSize * width * height];
120  Convert24Bit(pixelData, data, width, height);
121  }
122 
123  memset(&subresourceData, 0, sizeof(D3D11_SUBRESOURCE_DATA));
124  subresourceData.pSysMem = pixelData;
125  subresourceData.SysMemPitch = texelSize*width;
126 
127  subresourcePtr = &subresourceData;
128  }
129 
130  D3D11_TEXTURE2D_DESC desc;
131  memset(&desc, 0, sizeof(D3D11_TEXTURE2D_DESC));
132  desc.Width = width;
133  desc.Height = height;
134  desc.MipLevels = 1;
135  desc.ArraySize = 1;
136  desc.Format = DXGI_FORMAT_R8G8B8A8_UNORM_SRGB;
137  desc.SampleDesc.Count = 1;
138  desc.SampleDesc.Quality = 0;
139  switch (usage)
140  {
141  default:
142  case kDefaultUsage:
143  desc.Usage = D3D11_USAGE_DEFAULT;
144  break;
145  case kImmutableUsage:
146  desc.Usage = D3D11_USAGE_IMMUTABLE;
147  break;
148  case kDynamicUsage:
149  desc.Usage = D3D11_USAGE_DYNAMIC;
150  break;
151  }
152  desc.BindFlags = D3D11_BIND_SHADER_RESOURCE;
153  if (kDynamicUsage == usage)
154  {
155  //*** not sure if needed
156  desc.CPUAccessFlags = D3D11_CPU_ACCESS_WRITE;
157  }
158 
159  if (FAILED(device->CreateTexture2D(&desc, subresourcePtr, &mTexturePtr)))
160  {
161  return false;
162  }
163 
164  if (kRGB24TexFmt == format)
165  {
166  delete [] pixelData;
167  }
168 
169  mD3DFormat = DXGI_FORMAT_B8G8R8A8_UNORM_SRGB;
170  // verify formats
171 
172  if (FAILED(device->CreateShaderResourceView(mTexturePtr, nullptr, &mShaderResourceView)))
173  {
174  mTexturePtr->Release();
175  mTexturePtr = 0;
176  return false;
177  }
178 
179  return true;
180 }

182 //-------------------------------------------------------------------------------
183 // @ IvTextureD3D11::CreateMipmapped()
184 //-------------------------------------------------------------------------------
185 // Texture initialization
186 //-------------------------------------------------------------------------------
187 bool
188 IvTextureD3D11::CreateMipmapped(unsigned int width, unsigned int height, IvTextureFormat format,
189  void** data, unsigned int levels, IvDataUsage usage, ID3D11Device* device)
190 {
191  if (width == 0 || height == 0 || mTexturePtr)
192  {
193  return false;
194  }
195 
196  if (usage == kDynamicUsage)
197  {
198  // D3D11 doesn't allow creation of mipmapped dynamic textures
199  return false;
200  }
201 
202  if (usage == kImmutableUsage && !data)
203  {
204  return false;
205  }
206 
207  mWidth = width;
208  mHeight = height;
209  mFormat = format;
210  mUsage = usage;
211 
212  unsigned int texelSize = sInternalTextureFormatSize[mFormat];
213 
214  mLevelCount = 1;
215 
216  int size = texelSize * width * height;
217  while ((width != 1) || (height != 1))
218  {
219  width >>= 1;
220  height >>= 1;
221 
222  if (!width)
223  width = 1;
224 
225  if (!height)
226  height = 1;
227 
228  size += texelSize * width * height;
229  mLevelCount++;
230  }
231 
232  mLevels = new Level[mLevelCount];
233 
234  mLevels[0].mData = 0;
235  mLevels[0].mWidth = mWidth;
236  mLevels[0].mHeight = mHeight;
237  mLevels[0].mSize = mWidth * mHeight * texelSize;
238 
239  width = mWidth;
240  height = mHeight;
241 
242  unsigned int i;
243  for (i = 1; i < mLevelCount; i++)
244  {
245  width >>= 1;
246  height >>= 1;
247 
248  if (!width)
249  width = 1;
250 
251  if (!height)
252  height = 1;
253 
254  unsigned int size = texelSize * width * height;
255 
256  mLevels[i].mData = 0;
257  mLevels[i].mWidth = width;
258  mLevels[i].mHeight = height;
259  mLevels[i].mSize = size;
260  }
261 
262  D3D11_SUBRESOURCE_DATA* subresourceData = nullptr;
263  if (data)
264  {
265  subresourceData = new D3D11_SUBRESOURCE_DATA[mLevelCount];
266 
267  for (unsigned int level = 0; level < mLevelCount; ++level)
268  {
269  void* pixelData = data[level];
270 
271  unsigned int levelWidth = mLevels[level].mWidth;
272  unsigned int levelHeight = mLevels[level].mHeight;
273 
274  // if 24-bit color, we need to convert to 32-bit color
275  if (data && kRGB24TexFmt == format)
276  {
277  mLevels[level].mData = new unsigned char[texelSize * levelWidth * levelHeight];
278  Convert24Bit(mLevels[level].mData, data, width, height);
279  pixelData = mLevels[level].mData;
280  }
281 
282  memset(&subresourceData[level], 0, sizeof(D3D11_SUBRESOURCE_DATA));
283  subresourceData[level].pSysMem = pixelData;
284  subresourceData[level].SysMemPitch = texelSize*levelWidth;
285  }
286  }
287 
288  D3D11_TEXTURE2D_DESC desc;
289  memset(&desc, 0, sizeof(D3D11_TEXTURE2D_DESC));
290  desc.Width = mWidth;
291  desc.Height = mHeight;
292  desc.MipLevels = mLevelCount;
293  desc.ArraySize = 1;
294  desc.Format = DXGI_FORMAT_R8G8B8A8_UNORM_SRGB;
295  desc.SampleDesc.Count = 1;
296  desc.SampleDesc.Quality = 0;
297  switch (usage)
298  {
299  default:
300  case kDefaultUsage:
301  desc.Usage = D3D11_USAGE_DEFAULT;
302  break;
303  case kImmutableUsage:
304  desc.Usage = D3D11_USAGE_IMMUTABLE;
305  break;
306  }
307  desc.BindFlags = D3D11_BIND_SHADER_RESOURCE;
308 
309  if (FAILED(device->CreateTexture2D(&desc, subresourceData, &mTexturePtr)))
310  {
311  return false;
312  }
313 
314  if (data)
315  {
316  delete [] subresourceData;
317  if (kRGB24TexFmt == format)
318  {
319  for (unsigned int level = 0; level < mLevelCount; ++level)
320  {
321  delete[] mLevels[level].mData;
322  mLevels[level].mData = nullptr;
323  }
324  }
325  }
326 
327  mD3DFormat = DXGI_FORMAT_B8G8R8A8_UNORM_SRGB;
328  // verify formats
329 
330  if (FAILED(device->CreateShaderResourceView(mTexturePtr, nullptr, &mShaderResourceView)))
331  {
332  mTexturePtr->Release();
333  mTexturePtr = 0;
334  return false;
335  }
336 
337  return true;
338 }

412 void* IvTextureD3D11::BeginLoadData(unsigned int level)
413 {

429 //-------------------------------------------------------------------------------
430 // @ IvTextureD3D11::EndLoadData()
431 //-------------------------------------------------------------------------------
432 // Uploads the edited texture data to D3D
433 //-------------------------------------------------------------------------------
434 bool IvTextureD3D11::EndLoadData(unsigned int level)
435 {
436  if (kImmutableUsage == mUsage || level >= mLevelCount)
437  {
438  return false;
439  }
440 
441  // not already "locked"
442  if (!mLevels[level].mData)
443  {
444  return false;
445  }
446 
447  // this is seriously ugly -- not clear how to easily get the context down here
448  ID3D11DeviceContext* d3dContext = ((IvRendererD3D11*)IvRenderer::mRenderer)->GetContext();
449  if (kDefaultUsage == mUsage)
450  {
451  // use UpdateSubresource()
452 
453  void* pixelData = mLevels[level].mData;
454  if (kRGB24TexFmt == mFormat)
455  {
456  pixelData = new unsigned char[4 * mLevels[level].mWidth * mLevels[level].mHeight];
457  Convert24Bit(pixelData, mLevels[level].mData, mLevels[level].mWidth, mLevels[level].mHeight);
458  }
459 
460  d3dContext->UpdateSubresource(mTexturePtr, level, nullptr, pixelData, 4 * mLevels[level].mWidth, mLevels[level].mSize);
461 
462  if (kRGB24TexFmt == mFormat)
463  {
464  delete [] pixelData;
465  }
466  }
467  else if (kDynamicUsage == mUsage)
468  {
469  // D3D11 doesn't allow dynamic mipmapped textures
470  ASSERT(level == 0);
471 
472  // use Map/Unmap
473  D3D11_MAPPED_SUBRESOURCE mappedResource;
474  ZeroMemory(&mappedResource, sizeof(D3D11_MAPPED_SUBRESOURCE));
475  // Disable GPU access to the vertex buffer data.
476  if (S_OK != d3dContext->Map(mTexturePtr, level, D3D11_MAP_WRITE_DISCARD, 0, &mappedResource))
477  {
478  return false;
479  }
480 
481  void* pixelData = mappedResource.pData;
482  if (kRGB24TexFmt == mFormat)
483  {
484  Convert24Bit(pixelData, mLevels[level].mData, mLevels[level].mWidth, mLevels[level].mHeight);
485  }
486  else
487  {
488  memcpy(pixelData, mLevels[level].mData, mLevels[level].mSize);
489  }
490 
491  // Reenable GPU access to the vertex buffer data.
492  d3dContext->Unmap(mTexturePtr, level);
493  }
494 
495  delete [] mLevels[level].mData;
496  mLevels[level].mData = nullptr;
497 
498  return true;
499 }
