52 bool
53 IvTextureOGL::Create(unsigned int width, unsigned int height, IvTextureFormat format,
54  void* data, IvDataUsage usage)
55 {
56  if ( width == 0 || height == 0 || mID != 0 )
57  {
58  return false;
59  }
60 
61  if ( usage == kImmutableUsage && !data )
62  {
63  return false;
64  }
65 
66  glGenTextures(1, &mID);
67 
68  mWidth = width;
69  mHeight = height;
70  mFormat = format;
71 
72  mLevelCount = 1;
73 
74  GLint currentTex;
75  glGetIntegerv(GL_TEXTURE_BINDING_2D, &currentTex);
76 
77  glBindTexture(GL_TEXTURE_2D, mID);
78  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAX_LEVEL, mLevelCount-1);
79 
80  if (data)
81  {
82  width = mWidth;
83  height = mHeight;
84 
85  switch (mFormat)
86  {
87  case kRGBA32TexFmt:
88  glTexImage2D(GL_TEXTURE_2D, 0, GL_SRGB8_ALPHA8,
89  width, height, 0,
90  GL_RGBA, GL_UNSIGNED_BYTE, data);
91  break;
92 
93  case kRGB24TexFmt:
94  glTexImage2D(GL_TEXTURE_2D, 0, GL_SRGB8,
95  width, height, 0,
96  GL_RGB, GL_UNSIGNED_BYTE, data);
97  break;
98 
99  default:
100  break;
101  };
102  }
103 
104  glBindTexture(GL_TEXTURE_2D, currentTex);
105 
106  mUsage = usage;
107  mTempData = nullptr;
108 
109  return true;
110 }

118 IvTextureOGL::CreateMipmapped(unsigned int width, unsigned int height, IvTextureFormat format,
119  void** data, unsigned int levels, IvDataUsage usage)
120 {
121  if ( width == 0 || height == 0 || mID != 0 )
122  {
123  return false;
124  }
125 
126  if ( usage == kImmutableUsage && !data )
127  {
128  return false;
129  }
130 
131  glGenTextures(1, &mID);
132 
133  mWidth = width;
134  mHeight = height;
135  mFormat = format;
136 
137  //*** TODO: should we specify levels?
138  if (levels == 0)
139  {
140  mLevelCount = 1;
141  while ((width != 1) || (height != 1))
142  {
143  width >>= 1;
144  height >>= 1;
145 
146  if (!width)
147  width = 1;
148 
149  if (!height)
150  height = 1;
151 
152  mLevelCount++;
153  }
154  }
155  else
156  {
157  mLevelCount = levels;
158  }
159 
160  GLint currentTex;
161  glGetIntegerv(GL_TEXTURE_BINDING_2D, &currentTex);
162 
163  glBindTexture(GL_TEXTURE_2D, mID);
164  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAX_LEVEL, mLevelCount-1);
165 
166  if (data)
167  {
168  width = mWidth;
169  height = mHeight;
170 
171  void** dataPtr = data;
172 
173  unsigned int level;
174  for (level = 0; level < mLevelCount; level++)
175  {
176  switch (mFormat)
177  {
178  case kRGBA32TexFmt:
179  glTexImage2D(GL_TEXTURE_2D, level, GL_RGBA,
180  width, height, 0,
181  GL_RGBA, GL_UNSIGNED_BYTE, *dataPtr);
182  break;
183 
184  case kRGB24TexFmt:
185  glTexImage2D(GL_TEXTURE_2D, level, GL_RGB,
186  width, height, 0,
187  GL_RGB, GL_UNSIGNED_BYTE, *dataPtr);
188  break;
189 
190  default:
191  break;
192  };
193 
194  width >>= 1;
195  height >>= 1;
196 
197  if (!width)
198  width = 1;
199 
200  if (!height)
201  height = 1;
202 
203  ++dataPtr;
204  }
205  }
206 
207  glBindTexture(GL_TEXTURE_2D, currentTex);
208 
209  mUsage = usage;
210  mTempData = nullptr;
211 
212  return true;

245 // "Locks" and returns a pointer to the sysmem texture data for the given level
246 //-------------------------------------------------------------------------------
247 void* IvTextureOGL::BeginLoadData(unsigned int level)
248 {
249  if (mUsage == kImmutableUsage || mTempData)
250  {
251  return nullptr;
252  }
253 
254  unsigned width = mWidth;
255  unsigned height = mHeight;
256 
257  width >>= level;
258  height >>= level;
259 
260  if (!width)
261  width = 1;
262 
263  if (!height)
264  height = 1;
265 
266  unsigned int texelSize = sTextureFormatSize[mFormat];
267  unsigned int size = texelSize * width * height;
268 
269  mTempData = new unsigned char[size];
270 
271  return mTempData;
272 }
