207 //----------------------------------------------------------------------------
208 // @ ::DistanceSquared()
209 // ---------------------------------------------------------------------------
210 // Returns the distance squared between lines.
211 //-----------------------------------------------------------------------------
212 float DistanceSquared( const IvLine3& line0, const IvLine3& line1,
213  float& s_c, float& t_c )
214 {
215  IvVector3 w0 = line0.mOrigin - line1.mOrigin;
216  float a = line0.mDirection.Dot( line0.mDirection );
217  float b = line0.mDirection.Dot( line1.mDirection );
218  float c = line1.mDirection.Dot( line1.mDirection );
219  float d = line0.mDirection.Dot( w0 );
220  float e = line1.mDirection.Dot( w0 );
221  float denom = a*c - b*b;
222  if ( IvIsZero(denom) )
223  {
224  s_c = 0.0f;
225  t_c = e/c;
226  IvVector3 wc = w0 - t_c*line1.mDirection;
227  return wc.Dot(wc);
228  }
229  else
230  {
231  s_c = ((b*e - c*d)/denom);
232  t_c = ((a*e - b*d)/denom);
233  IvVector3 wc = w0 + s_c*line0.mDirection
234  - t_c*line1.mDirection;
235  return wc.Dot(wc);
236  }
237 
238 } // End of ::DistanceSquared()

241 //----------------------------------------------------------------------------
242 // @ ::DistanceSquared()
243 // ---------------------------------------------------------------------------
244 // Returns the distance squared between line and point.
245 //-----------------------------------------------------------------------------
246 float DistanceSquared( const IvLine3& line, const IvVector3& point, float& t_c )
247 {
248  IvVector3 w = point - line.mOrigin;
249  float vsq = line.mDirection.Dot(line.mDirection);
250  float proj = w.Dot(line.mDirection);
251  t_c = proj/vsq;
252 
253  return w.Dot(w) - t_c*proj;
254 
255 } // End of ::DistanceSquared()

261 // Returns the closest points between two lines
262 //-----------------------------------------------------------------------------
263 void ClosestPoints( IvVector3& point0, IvVector3& point1,
264  const IvLine3& line0,
265  const IvLine3& line1 )
266 {
267  // compute intermediate parameters
268  IvVector3 w0 = line0.mOrigin - line1.mOrigin;
269  float a = line0.mDirection.Dot( line0.mDirection );
270  float b = line0.mDirection.Dot( line1.mDirection );
271  float c = line1.mDirection.Dot( line1.mDirection );
272  float d = line0.mDirection.Dot( w0 );
273  float e = line1.mDirection.Dot( w0 );
274 
275  float denom = a*c - b*b;
276 
277  if ( IvIsZero(denom) )
278  {
279  point0 = line0.mOrigin;
280  point1 = line1.mOrigin + (e/c)*line1.mDirection;
281  }
282  else

284  point0 = line0.mOrigin + ((b*e - c*d)/denom)*line0.mDirection;
285  point1 = line1.mOrigin + ((a*e - b*d)/denom)*line1.mDirection;
