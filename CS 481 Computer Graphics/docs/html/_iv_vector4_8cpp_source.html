206 //-------------------------------------------------------------------------------

208 //-------------------------------------------------------------------------------
209 // Set to unit vector
210 //-------------------------------------------------------------------------------
211 void
212 IvVector4::Normalize()
213 {
214  float lengthsq = x*x + y*y + z*z + w*w;
215 
216  if (IvIsZero(lengthsq))
217  {
218  Zero();
219  }
220  else
221  {
222  float factor = IvRecipSqrt(lengthsq);
223  x *= factor;
224  y *= factor;
225  z *= factor;
226  w *= factor;
