260 void IvPointRendererD3D11::Setup()
261 {
262  sVertexShader[kPFormat] = sVertexShaderPFormat;
263  sVertexShader[kCPFormat] = sVertexShaderCPFormat;
264  sVertexShader[kNPFormat] = sVertexShaderNPFormat;
265  sVertexShader[kCNPFormat] = sVertexShaderCNPFormat;
266  sVertexShader[kTCPFormat] = sVertexShaderTCPFormat;
267  sVertexShader[kTNPFormat] = sVertexShaderTNPFormat;
268 
269  sFragmentShader[kPFormat] = sFragmentShaderP;
270  sFragmentShader[kCPFormat] = sFragmentShaderCP;
271  sFragmentShader[kNPFormat] = sFragmentShaderCP;
272  sFragmentShader[kCNPFormat] = sFragmentShaderCP;
273  sFragmentShader[kTCPFormat] = sFragmentShaderTCP;
274  sFragmentShader[kTNPFormat] = sFragmentShaderTCP;
275 
276  for (int format = 0; format < kVertexFormatCount; ++format)
277  {
278  // create the shaders
279  sShaders[format] = IvRenderer::mRenderer->GetResourceManager()->CreateShaderProgram(
280  IvRenderer::mRenderer->GetResourceManager()->CreateVertexShaderFromString(
281  sVertexShader[format]),
282  IvRenderer::mRenderer->GetResourceManager()->CreateFragmentShaderFromString(
283  sFragmentShader[format]));
284 
285  // create the vertex layout
286  D3D11_INPUT_ELEMENT_DESC* elements = nullptr;
287  UINT numElements = 0;
288  const char* shaderString = nullptr;
289 
290  switch (format)
291  {
292  case kPFormat:
293  elements = sPFormatElements;
294  numElements = sizeof(sPFormatElements) / sizeof(D3D11_INPUT_ELEMENT_DESC);
295  shaderString = sVertexShaderPFormat;
296  break;
297  case kCPFormat:
298  elements = sCPFormatElements;
299  numElements = sizeof(sCPFormatElements) / sizeof(D3D11_INPUT_ELEMENT_DESC);
300  shaderString = sVertexShaderCPFormat;
301  break;
302  case kNPFormat:
303  elements = sNPFormatElements;
304  numElements = sizeof(sNPFormatElements) / sizeof(D3D11_INPUT_ELEMENT_DESC);
305  shaderString = sVertexShaderNPFormat;
306  break;
307  case kCNPFormat:
308  elements = sCNPFormatElements;
309  numElements = sizeof(sCNPFormatElements) / sizeof(D3D11_INPUT_ELEMENT_DESC);
310  shaderString = sVertexShaderCNPFormat;
311  break;
312  case kTCPFormat:
313  elements = sTCPFormatElements;
314  numElements = sizeof(sTCPFormatElements) / sizeof(D3D11_INPUT_ELEMENT_DESC);
315  shaderString = sVertexShaderTCPFormat;
316  break;
317  case kTNPFormat:
318  elements = sTNPFormatElements;
319  numElements = sizeof(sTNPFormatElements) / sizeof(D3D11_INPUT_ELEMENT_DESC);
320  shaderString = sVertexShaderTNPFormat;
321  break;
322  }
323 
324  ID3DBlob* code;
325  ID3DBlob* errorMessages = nullptr;
326 
327  // compile the shader to assembly
328  // it's not ideal to compile the vertex shader again, but we only have to
329  // go through this process once to create the layout
330  if (FAILED(D3DCompile(shaderString, strlen(shaderString) + 1, nullptr, nullptr, nullptr, "vs_main", "vs_4_0",
331  D3DCOMPILE_ENABLE_STRICTNESS | D3DCOMPILE_DEBUG, 0, &code, &errorMessages)))
332  {
333  const char* errors = reinterpret_cast<const char*>(errorMessages->GetBufferPointer());
334  DEBUG_OUT("Vertex shader error: ");
335  DEBUG_OUT(errors << std::endl);
336  errorMessages->Release();
337  ASSERT(false);
338  return;
339  }
340  if (errorMessages)
341  {
342  errorMessages->Release();
343  }
344 
345  ID3D11Device* device = static_cast<IvRendererD3D11*>(IvRenderer::mRenderer)->GetDevice();
346  if (FAILED(device->CreateInputLayout(elements, numElements, code->GetBufferPointer(), code->GetBufferSize(), &sInputLayout[format])))
347  {
348  code->Release();
349  ASSERT(false);
350  return;
351  }
352  code->Release();
353  }
354 
355  // create the quad buffer
356  float xCoord = 1.0f / static_cast<float>(IvRenderer::mRenderer->GetWidth());
357  float yCoord = 1.0f / static_cast<float>(IvRenderer::mRenderer->GetHeight());
358  IvPVertex verts[4] =
359  {
360  IvVector3(xCoord, -yCoord, 0.0f),
361  IvVector3(xCoord, yCoord, 0.0f),
362  IvVector3(-xCoord, -yCoord, 0.0f),
363  IvVector3(-xCoord, yCoord, 0.0f)
364  };
365  sQuadBuffer = IvRenderer::mRenderer->GetResourceManager()->CreateVertexBuffer(kPFormat, 4, verts, kImmutableUsage);
366 }

383 void IvPointRendererD3D11::SetShaderAndVertexBuffer(ID3D11DeviceContext* context,
384  IvVertexFormat format, IvVertexBuffer* buffer)
385 {
386  IvRenderer::mRenderer->SetShaderProgram(sShaders[format]);
387  sShaders[format]->GetUniform("pointScale")->SetValue(sPointSize, 0);
388 
389  ID3D11Buffer* instancePtr = reinterpret_cast<IvVertexBufferD3D11*>(buffer)->GetBufferPtr();
390  ID3D11Buffer* quadPtr = reinterpret_cast<IvVertexBufferD3D11*>(sQuadBuffer)->GetBufferPtr();
391  ID3D11Buffer* buffers[2] = { instancePtr, quadPtr };
392  UINT strides[2] = { kIvVFSize[format], kIvVFSize[kPFormat] };
393  UINT offsets[2] = { 0, 0 };
394 
395  context->IASetVertexBuffers(0, 2, buffers, strides, offsets);
396  context->IASetInputLayout(sInputLayout[format]);
397 }
